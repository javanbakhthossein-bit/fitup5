   654→function ChargeWalletDialog({ user, onClose, onDone }: { user: any; onClose: () => void; onDone: () => void }) {

   656→  const [desc, setDesc] = useState("");

   668→      toast.success(`کیف پول ${toPersianDigits(formatToman(Number(amount)))} ت شارژ شد`);

   670→    } catch { toast.error("خطا در شارژ"); } finally { setSaving(false); }
