   115→   115→// بر اساس فرمول‌های استاندارد Harris-Benedict و Mifflin-St Jeor
   116→   116→function getActivityMultiplier(activityLevel?: OnboardingData["activityLevel"]): number {
   117→   117→  switch (activityLevel) {
   118→   118→    case "sedentary":
   119→   119→      return 1.2;   // بی‌تحرک (کار پشت میز، بدون تمرین)
   120→   120→    case "light":
   121→   121→      return 1.375; // کم‌تحرک (تمرین ۱-۳ روز در هفته)
   122→   122→    case "moderate":
   123→   123→      return 1.55;  // متوسط (تمرین ۳-۵ روز در هفته)
   124→   124→    case "active":
   125→   125→      return 1.725; // فعال (تمرین ۶-۷ روز در هفته)
   126→   126→    case "very_active":
   127→   127→      return 1.9;   // خیلی فعال (تمرین سنگین روزانه + کار فیزیکی)
   128→   128→    default:
   129→   129→      return 1.4;   // پیش‌فرض محافظه‌کارانه
   130→   130→  }

   241→   241→// ─── محاسبه دقیق TDEE و کالری هدف بر اساس هدف و فعالیت ───
   242→   242→function computeTDEEAndTarget(data: OnboardingData): {
   243→   243→  bmr: number;
   244→   244→  tdee: number;
   245→   245→  targetCalories: number;
   246→   246→  calorieAdjustment: number;
   247→   247→  proteinG: number;
   248→   248→  carbsG: number;
   249→   249→  fatG: number;
   250→   250→  proteinPerKg: number;
   251→   251→  carbsPerKg: number;
   252→   252→  fatPerKg: number;
   253→   253→} {
   254→   254→  // فرمول Mifflin-St Jeor (دقیق‌ترین فرمول BMR)
   255→   255→  const bmr =
   256→   256→    data.gender === "male"
   257→   257→      ? 10 * data.weight + 6.25 * data.height - 5 * data.age + 5
   258→   258→      : 10 * data.weight + 6.25 * data.height - 5 * data.age - 161;
   259→   259→  const activityMultiplier = getActivityMultiplier(data.activityLevel);
   260→   260→  const tdee = bmr * activityMultiplier;
   261→   261→
   262→   262→  // تنظیم کالری بر اساس هدف — مازاد/نقصان هوشمند
   263→   263→  let calorieAdjustment = 0;
   264→   264→  switch (data.goal) {
   265→   265→    case "fat_loss":
   266→   266→      // نقصان ۴۰۰-۵۰۰ کالری برای چربی‌سوزی ایمن (۰.۵ کیلو در هفته)
   267→   267→      calorieAdjustment = -Math.min(500, Math.round(tdee * 0.2));
   268→   268→      break;
   269→   269→    case "muscle_gain":
   270→   270→      // مازاد ۲۵۰-۴۰۰ کالری برای حجم‌گیری تمیز (به حداقل رساندن چربی‌سازی)
   271→   271→      calorieAdjustment = 300;
   272→   272→      break;
   273→   273→    case "strength":
   274→   274→      // مازاد ملایم ۲۰۰ کالری برای قدرت
   275→   275→      calorieAdjustment = 200;
   276→   276→      break;
   277→   277→    case "endurance":
   278→   278→      // تعادل با کمی مازاد برای ریکاوری
   279→   279→      calorieAdjustment = 100;
   280→   280→      break;
   281→   281→    case "fitness":
   282→   282→    default:
   283→   283→      calorieAdjustment = 0;
   284→   284→      break;
   285→   285→  }
   286→   286→  const targetCalories = Math.round(tdee + calorieAdjustment);
   287→   287→
   288→   288→  // محاسبه درشت‌مغذی‌ها — پروتئین بر اساس هدف، چربی ۲۵٪، کربوهیدرات باقی‌مانده
   289→   289→  const proteinPerKg = data.goal === "fat_loss" ? 2.2 : data.goal === "muscle_gain" ? 2.0 : 1.8;
   290→   290→  const proteinG = Math.round(data.weight * proteinPerKg);
   291→   291→  const fatG = Math.round((targetCalories * 0.25) / 9); // ۲۵٪ کالری از چربی
   292→   292→  const carbsG = Math.max(0, Math.round((targetCalories - proteinG * 4 - fatG * 9) / 4));
   293→   293→  const carbsPerKg = Math.round((carbsG / data.weight) * 10) / 10;
   294→   294→  const fatPerKg = Math.round((fatG / data.weight) * 10) / 10;
   295→   295→
   296→   296→  return {
   297→   297→    bmr: Math.round(bmr),
   298→   298→    tdee: Math.round(tdee),
   299→   299→    targetCalories,
   300→   300→    calorieAdjustment,
   301→   301→    proteinG,
   302→   302→    carbsG,
   303→   303→    fatG,
   304→   304→    proteinPerKg,
   305→   305→    carbsPerKg,
   306→   306→    fatPerKg,
   307→   307→  };
   308→   308→}
