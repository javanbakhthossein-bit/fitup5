   100→function log(
   101→  ctx: RunContext,
   102→  level: RunProgress["level"],
   103→  msg: string
   104→): void {
   105→  const entry: RunProgress = { level, msg, ts: Date.now() };
   106→  ctx.logs.push(entry);
   107→  // Also print to server log for debugging
   108→  const prefix =
   109→    level === "error"
   110→      ? "[SEO-AGENT:ERROR]"
   111→      : level === "warn"
   112→      ? "[SEO-AGENT:WARN]"
   113→      : level === "success"
   114→      ? "[SEO-AGENT:OK]"
   115→      : "[SEO-AGENT]";
   116→  console.log(`${prefix} ${msg}`);
   117→  // Persist to DB every 5 logs or every 3 seconds — for live UI updates
   118→  const now = Date.now();
   119→  const lastPersist = (ctx as any)._lastPersist || 0;
   120→  if (ctx.logs.length % 5 === 0 || now - lastPersist > 3000) {
   121→    (ctx as any)._lastPersist = now;
   122→    // fire-and-forget — don't block the agent
   123→    db.seoAgentRun
   124→      .update({
   125→        where: { id: ctx.runId },
   126→        data: {
   127→          logs: JSON.stringify(ctx.logs.slice(-200)),
   128→          successCount: ctx.successCount,
   129→          failCount: ctx.failCount,
   130→        },
   131→      })
   132→      .catch(() => null);
   133→  }
   134→}

   136→// ─── JSON extraction helper ───
   137→function extractJson(content: string): any | null {
   138→  if (!content) return null;
   139→  // Try code block first
   140→  const codeBlock = content.match(/```(?:json)?\s*([\s\S]*?)```/);
   141→  const candidate = codeBlock ? codeBlock[1].trim() : content.trim();
   142→  // Find the first balanced { ... }
   143→  const start = candidate.indexOf("{");
   144→  if (start === -1) return null;
   145→  let depth = 0;
   146→  for (let i = start; i < candidate.length; i++) {
   147→    if (candidate[i] === "{") depth++;
   148→    else if (candidate[i] === "}") {
   149→      depth--;
   150→      if (depth === 0) {
   151→        const jsonStr = candidate.slice(start, i + 1);
   152→        try {
   153→          return JSON.parse(jsonStr);
   154→        } catch {
   155→          // try to fix common issues
   156→          try {
   157→            return JSON.parse(
   158→              jsonStr
   159→                .replace(/,\s*}/g, "}")
   160→                .replace(/,\s*]/g, "]")
   161→                .replace(/[\u201C\u201D]/g, '"')
   162→                .replace(/[\u2018\u2019]/g, "'")
   163→            );
   164→          } catch {
   165→            return null;
   166→          }
   167→        }
   168→      }
   169→    }
   170→  }
   171→  return null;
   172→}

   174→async function callLlm(system: string, user: string, maxTokens = 4000): Promise<string> {
   175→  // timeout ۳ دقیقه — برای production که AI کندتر پاسخ می‌دهد
   176→  const timeoutMs = 180000;
   177→  const controller = new AbortController();
   178→  const timeout = setTimeout(() => controller.abort(), timeoutMs);
   179→  try {
   180→    const completion = await getAvalaiClient().chat.completions.create({
   181→      model: TEXT_MODEL,
   182→      messages: [
   183→        { role: "system", content: system },
   184→        { role: "user", content: user },
   185→      ],
   186→      max_tokens: maxTokens,
   187→      temperature: 0.7,
   188→      // @ts-expect-error — OpenAI SDK supports signal
   189→      signal: controller.signal,
   190→    } as any);
   191→    return completion.choices[0]?.message?.content || "";
   192→  } catch (e: any) {
   193→    if (e.name === "AbortError") {
   194→      throw new Error(`AI timeout after ${timeoutMs / 1000}s`);
   195→    }
   196→    throw e;
   197→  } finally {
   198→    clearTimeout(timeout);
   199→  }
   200→}

   203→export async function analyzeSite(ctx: RunContext): Promise<SiteAnalysis> {
   204→  log(ctx, "info", "🔍 شروع تحلیل سایت — استخراج صفحات و مقالات موجود");
   205→  // Static landing pages & tools (همیشه در سایت فیتاپ وجود دارند)
   206→  // مهم: همه مسیرهای معتبر لینک داخلی در سایت فیتاپ. از این لیست برای لینک‌سازی استفاده کن.
   207→  const pages = [
   208→    { url: "/", title: "فیتاپ — مربی هوشمند بدنسازی و تغذیه", type: "landing" as const },
   209→    { url: "/?screen=articles", title: "مقالات ورزشی و تغذیه فیتاپ", type: "list" as const },
   210→    { url: "/?tool=tdee", title: "محاسبه‌گر کالری و TDEE", type: "tool" as const },
   211→    { url: "/?tool=exercises", title: "بانک حرکات ورزشی", type: "tool" as const },
   212→    { url: "/?tool=foods", title: "جدول کالری غذاها", type: "tool" as const },
   213→    { url: "/?screen=auth", title: "ورود / ثبت‌نام فیتاپ", type: "landing" as const },
   214→  ];
   215→
   216→  const articles = await db.article.findMany({
   217→    where: { status: "published" },
   218→    select: {
   219→      id: true,
   220→      title: true,
   221→      slug: true,
   222→      category: true,
   223→      tags: true,
   224→      excerpt: true,
   225→    },
   226→    orderBy: { createdAt: "desc" },
   227→  });
   228→
   229→  // Collect existing keywords from tags
   230→  const existingKeywords = new Set<string>();
   231→  for (const a of articles) {
   232→    if (a.tags) {
   233→      for (const t of a.tags.split(",").map((s) => s.trim()).filter(Boolean)) {
   234→        existingKeywords.add(t);
   235→      }
   236→    }
   237→  }
   238→
   239→  const totalViewsAgg = await db.article.aggregate({
   240→    where: { status: "published" },
   241→    _sum: { views: true },
   242→  });
   243→
   244→  const result: SiteAnalysis = {
   245→    pages,
   246→    articles: articles.map((a) => ({
   247→      id: a.id,
   248→      title: a.title,
   249→      slug: a.slug,
   250→      category: a.category,
   251→      tags: a.tags,
   252→      excerpt: a.excerpt,
   253→    })),
   254→    existingKeywords: Array.from(existingKeywords),
   255→    totalArticles: articles.length,
   256→    totalViews: totalViewsAgg._sum.views || 0,
   257→  };
   258→
   259→  log(
   260→    ctx,
   261→    "success",
   262→    `✅ تحلیل سایت کامل شد: ${result.pages.length} صفحه کلیدی، ${result.totalArticles} مقاله موجود، ${result.existingKeywords.length} کلمه کلیدی موجود، ${result.totalViews} بازدید کل`
   263→  );
   264→  return result;
   265→}

   802→/**
   803→ * Compute the scheduled publish date for a new article.
   804→ *
   805→ * Scheduling rules:
   806→ *  1. If the strategy's contentCalendar has an entry whose keyword matches
   807→ *     (case-insensitive substring) the article's plan keyword, the article is
   808→ *     scheduled for `now + (week * 7 days)`. Week 1 = next week, week 2 = 2
   809→ *     weeks from now, etc. — this spreads articles across the calendar as the
   810→ *     strategy intends.
   811→ *  2. Otherwise, the article is scheduled 7 days after the latest already-scheduled
   812→ *     draft article (so articles never collide on the same day). If no draft is
   813→ *     currently scheduled, it's scheduled 1 day from now as a sensible default.
   814→ *  3. The returned date is always strictly in the future (>= now + 1 hour) so the
   815→ *     cron publisher won't immediately fire on the same run.
   816→ *
   817→ * Returns null when `publishImmediately` is true (caller should publish now).
   818→ */
   819→export async function computeScheduledAt(
   820→  plan: ArticlePlan,
   821→  strategy: SeoStrategyContent | null | undefined,
   822→  publishImmediately: boolean = false
   823→): Promise<Date | null> {
   824→  if (publishImmediately) return null;
   825→
   826→  const now = new Date();
   827→  const oneHourFromNow = new Date(now.getTime() + 60 * 60 * 1000);
   828→
   829→  // 1) پیدا کردن آخرین مقاله زمان‌بندی‌شده (draft با scheduledAt)
   830→  const latestScheduled = await db.article.findFirst({
   831→    where: {
   832→      status: "draft",
   833→      scheduledAt: { not: null },
   834→    },
   835→    orderBy: { scheduledAt: "desc" },
   836→    select: { scheduledAt: true },
   837→  });
   838→
   839→  // 2) محاسبه تاریخ پایه
   840→  // اگر مقاله زمان‌بندی‌شده‌ای وجود دارد، از آن استفاده کن
   841→  // در غیر این صورت، از امروز استفاده کن
   842→  let baseDate = now;
   843→  if (latestScheduled?.scheduledAt) {
   844→    baseDate = new Date(latestScheduled.scheduledAt);
   845→  }
   846→
   847→  // 3) فاصله دقیقاً ۴ روز بین هر مقاله
   848→  // این تضمین می‌کند که مقالات با فاصله منظم ۴ روز منتشر شوند
   849→  // مثلاً ۱۰ مقاله = ۴۰ روز انتشار
   850→  const gapDays = 4;
   851→
   852→  const target = new Date(baseDate);
   853→  target.setDate(target.getDate() + gapDays);
   854→  target.setHours(9, 0, 0, 0); // ساعت ۹ صبح
   855→
   856→  // 4) مطمئن شو که تاریخ در آینده است
   857→  if (target.getTime() < oneHourFromNow.getTime()) {
   858→    // اگر تاریخ محاسبه‌شده در گذشته است، ۲ روز از امروز استفاده کن
   859→    const fallback = new Date(now);
   860→    fallback.setDate(fallback.getDate() + 2);
   861→    fallback.setHours(9, 0, 0, 0);
   862→    return fallback;
   863→  }
   864→
   865→  return target;
   866→}

  1189→با استفاده از اصولی که در این مقاله توضیح دادیم و با کمک مربی هوشمند فیتاپ، می‌توانید ${plan.keyword} را به‌صورت شخصی‌سازی‌شده در برنامه خود داشته باشید. کافیست برنامه اختصاصی فیتاپ را تهیه کنید.
  1190→
  1191→### آیا ${plan.keyword} برای مبتدی‌ها مناسب است؟
  1192→
  1193→بله — با رعایت نکات ایمنی و شروع تدریجی، ${plan.keyword} برای ورزشکاران مبتدی هم مناسب است. توصیه می‌کنیم قبل از شروع، با مربی هوشمند فیتاپ مشورت کنید تا برنامه‌ای متناسب با سطح شما طراحی شود.
  1194→`;
  1195→    articleContent = articleContent.trimEnd() + "\n" + faqSection;
  1196→    log(ctx, "info", `📋 بخش FAQ به مقاله اضافه شد (برای FAQPage JSON-LD).`);
  1197→  }
  1198→
  1199→  // ─── Compute reading time ───
  1200→  const readingMinutes = Math.max(1, Math.ceil(wordCount / 200));
  1201→
  1202→  // ─── Compute scheduled publish date (space out over weeks per strategy calendar) ───
  1203→  // The article is saved as "draft" with a scheduledAt in the future. The
  1204→  // /api/cron/publish-scheduled endpoint will flip it to "published" when the
  1205→  // scheduledAt arrives. This gives a steady drip of fresh content rather than
  1206→  // dumping N articles on the same day.
  1207→  const scheduledAt = await computeScheduledAt(plan, strategy, ctx.publishImmediately === true);
  1208→
  1209→  // ─── SEO fields: canonical URL, OG image, robots ───
  1210→  // canonical URL: همیشه مطلق (https://...) باشد تا گوگل آن را به‌عنوان canonical
  1211→  // صفحه قبول کند. در غیر این صورت، مقاله‌های تولیدشده canonical خالی دارند و
  1212→  // article-page.tsx به‌صورت fallback از ?article=slug استفاده می‌کند — اما
  1213→  // ذخیره URL مطلق در DB تمرین بهتری است و با مقاله‌های دستی هم‌خوانی دارد.
  1214→  const siteUrl = (process.env.NEXT_PUBLIC_SITE_URL || "https://fittup.ir").replace(/\/$/, "");
  1215→  const canonicalUrl = `${siteUrl}/?article=${plan.slug}`;
  1216→  const ogImage = coverUrl; // OG image همان تصویر کاور است
  1217→  const robots = "index,follow";
  1218→
  1219→  // ─── Save article to DB ───
  1220→  const article = await db.article.create({
  1221→    data: {
  1222→      title: plan.title,
  1223→      slug: plan.slug,
  1224→      excerpt: plan.excerpt,
  1225→      content: articleContent,
  1226→      category: plan.category,
  1227→      tags: [plan.keyword, ...plan.secondaryKeywords].slice(0, 8).join(", "),
  1228→      // If a schedule was computed → keep draft + scheduledAt; otherwise publish now.
  1229→      status: scheduledAt ? "draft" : "published",
  1230→      scheduledAt: scheduledAt ?? undefined,
  1231→      authorId: ctx.adminId,
  1232→      coverImage: coverUrl,
  1233→      seoTitle: seo.seoTitle,
  1234→      seoDescription: seo.seoDescription,
  1235→      metaKeywords: seo.metaKeywords,
  1236→      ogImage,
  1237→      canonicalUrl,
  1238→      robots,
  1239→      readingMinutes,
  1240→    },
  1241→  });
  1242→
  1243→  // Update plan as published
  1244→  if (plan.id) {
  1245→    await db.seoArticlePlan.update({
  1246→      where: { id: plan.id },
  1247→      data: { status: "published", articleId: article.id },
  1248→    });
  1249→  }
  1250→
  1251→  if (scheduledAt) {
  1252→    log(
  1253→      ctx,
  1254→      "success",
  1255→      `🗓 مقاله زمان‌بندی شد: "${plan.title}" — ${wordCount} کلمه، ${readingMinutes} دقیقه مطالعه، ${coverUrl ? "با تصویر" : "بدون تصویر"} — انتشار: ${scheduledAt.toLocaleDateString("fa-IR")} ساعت ${scheduledAt.toLocaleTimeString("fa-IR", { hour: "2-digit", minute: "2-digit" })}`
  1256→    );
  1257→  } else {
  1258→    log(
  1259→      ctx,
  1260→      "success",
  1261→      `✅ مقاله منتشر شد: "${plan.title}" — ${wordCount} کلمه، ${readingMinutes} دقیقه مطالعه، ${coverUrl ? "با تصویر" : "بدون تصویر"}`
  1262→    );
  1263→  }
  1264→
  1265→  return { articleId: article.id, coverUrl, wordCount, scheduledAt };
  1266→}

  1268→// ─── Main entry point: run the agent ───
  1269→export async function runSeoAgent(opts: {
  1270→  runId: string;
  1271→  adminId: string;
  1272→  mode: "full" | "continue" | "strategy_only";
  1273→  count: number;
  1274→  /**
  1275→   * If true, generated articles are published immediately instead of being
  1276→   * scheduled for future publishing. Defaults to false (schedule articles).
  1277→   */
  1278→  publishImmediately?: boolean;
  1279→}): Promise<RunContext> {
  1280→  const ctx: RunContext = {
  1281→    runId: opts.runId,
  1282→    mode: opts.mode,
  1283→    count: opts.count,
  1284→    logs: [],
  1285→    adminId: opts.adminId,
  1286→    startedAt: Date.now(),
  1287→    successCount: 0,
  1288→    failCount: 0,
  1289→    articles: [],
  1290→    errors: [],
  1291→    publishImmediately: opts.publishImmediately === true,
  1292→  };
  1293→
  1294→  const startedRun = await db.seoAgentRun.update({
  1295→    where: { id: opts.runId },
  1296→    data: {
  1297→      status: "running",
  1298→      mode: opts.mode,
  1299→      requestedCount: opts.count,
  1300→      startedAt: new Date(),
  1301→    },
  1302→  }).catch(() => null);
  1303→
  1304→  try {
  1305→    // Step 1: Always analyze site first (for accurate internal linking)
  1306→    const site = await analyzeSite(ctx);
  1307→
  1308→    // Step 2: Generate/refresh strategy (in full mode) or load existing
  1309→    let strategy: SeoStrategyContent;
  1310→    if (opts.mode === "full") {
  1311→      strategy = await generateStrategy(ctx, site);
  1312→    } else {
  1313→      const existing = await db.seoStrategy.findFirst({
  1314→        where: { isActive: true },
  1315→        orderBy: { version: "desc" },
  1316→      });
  1317→      if (existing) {
  1318→        strategy = JSON.parse(existing.content) as SeoStrategyContent;
  1319→        log(ctx, "info", `📋 استفاده از استراتژی موجود (نسخه ${existing.version})`);
  1320→      } else {
  1321→        // No strategy yet — must generate
  1322→        log(ctx, "warn", "استراتژی موجود نیست — تولید استراتژی جدید");
  1323→        strategy = await generateStrategy(ctx, site);
  1324→      }
  1325→    }
  1326→
  1327→    if (opts.mode === "strategy_only") {
  1328→      log(ctx, "success", "🏁 حالت strategy_only — استراتژی تولید شد، بدون تولید مقاله");
  1329→      await finishRun(ctx, "completed");
  1330→      return ctx;
  1331→    }
  1332→
  1333→    // Step 3: Plan articles (or use existing planned ones)
  1334→    let plans: ArticlePlan[] = [];
  1335→
  1336→    // First: use existing planned articles (continue mode)
  1337→    const existingPlans = await db.seoArticlePlan.findMany({
  1338→      where: { status: "planned" },
  1339→      orderBy: { priority: "desc" },
  1340→      take: opts.count,
  1341→    });
  1342→
  1343→    if (existingPlans.length >= opts.count && opts.mode === "continue") {
  1344→      plans = existingPlans.map((p) => ({
  1345→        id: p.id,
  1346→        keyword: p.keyword,
  1347→        title: p.title,
  1348→        slug: p.slug,
  1349→        category: p.category,
  1350→        excerpt: p.excerpt,
  1351→        pillar: p.pillar,
  1352→        secondaryKeywords: p.secondaryKeywords.split(", ").filter(Boolean),
  1353→        internalLinks: JSON.parse(p.internalLinks || "[]"),
  1354→        coverImagePrompt: p.coverImagePrompt,
  1355→        inlineImagePrompts: [],
  1356→      }));
  1357→      log(ctx, "info", `📋 استفاده از ${plans.length} مقاله برنامه‌ریزی‌شده موجود در صف`);
  1358→    } else {
  1359→      // Plan new articles
  1360→      const needed = opts.count - existingPlans.length;
  1361→      if (needed > 0) {
  1362→        log(ctx, "info", `📅 نیاز به ${needed} مقاله جدید — برنامه‌ریزی...`);
  1363→        await planArticles(ctx, strategy, site, Math.max(needed, opts.count));
  1364→      }
  1365→      // Now fetch all planned
  1366→      const allPlanned = await db.seoArticlePlan.findMany({
  1367→        where: { status: "planned" },
  1368→        orderBy: { priority: "desc" },
  1369→        take: opts.count,
  1370→      });
  1371→      plans = allPlanned.map((p) => ({
  1372→        id: p.id,
  1373→        keyword: p.keyword,
  1374→        title: p.title,
  1375→        slug: p.slug,
  1376→        category: p.category,
  1377→        excerpt: p.excerpt,
  1378→        pillar: p.pillar,
  1379→        secondaryKeywords: p.secondaryKeywords.split(", ").filter(Boolean),
  1380→        internalLinks: JSON.parse(p.internalLinks || "[]"),
  1381→        coverImagePrompt: p.coverImagePrompt,
  1382→        inlineImagePrompts: [],
  1383→      }));
  1384→    }
  1385→
  1386→    if (plans.length === 0) {
  1387→      log(ctx, "warn", "هیچ مقاله‌ای برای تولید یافت نشد");
  1388→      await finishRun(ctx, "completed");
  1389→      return ctx;
  1390→    }
  1391→
  1392→    log(ctx, "info", `🚀 شروع تولید ${plans.length} مقاله`);
  1393→
  1394→    // Step 4: Generate each article (sequentially to avoid rate limits)
  1395→    // Re-analyze site before each article to ensure accurate internal linking (new articles get added)
  1396→    for (let i = 0; i < plans.length; i++) {
  1397→      const plan = plans[i];
  1398→      log(ctx, "info", `━━━ مقاله ${i + 1}/${plans.length} ━━━`);
  1399→      try {
  1400→        // Refresh site analysis (so newly-published articles can be linked)
  1401→        const freshSite = i === 0 ? site : await analyzeSite(ctx);
  1402→        const result = await generateArticle(ctx, plan, freshSite, strategy);
  1403→        ctx.articles.push({
  1404→          id: result.articleId,
  1405→          title: plan.title,
  1406→          slug: plan.slug,
  1407→          keyword: plan.keyword,
  1408→          coverImage: result.coverUrl ? "✓" : "✗",
  1409→          words: result.wordCount,
  1410→          scheduledAt: result.scheduledAt ? result.scheduledAt.toISOString() : null,
  1411→        });
  1412→        ctx.successCount++;
  1413→      } catch (e: any) {
  1414→        log(ctx, "error", `مقاله "${plan.title}" شکست خورد: ${e.message}`);
  1415→        ctx.errors.push(`${plan.title}: ${e.message}`);
  1416→        ctx.failCount++;
  1417→        if (plan.id) {
  1418→          await db.seoArticlePlan.update({
  1419→            where: { id: plan.id },
  1420→            data: { status: "failed", errorMessage: e.message.slice(0, 500) },
  1421→          }).catch(() => null);
  1422→        }
  1423→      }
  1424→      // Persist progress to DB after each article
  1425→      await persistRunProgress(ctx);
  1426→    }
  1427→
  1428→    log(
  1429→      ctx,
  1430→      "success",
  1431→      `🏁 پایان اجرا — ${ctx.successCount} موفق، ${ctx.failCount} ناموفق`
  1432→    );
  1433→
  1434→    await finishRun(ctx, ctx.failCount === 0 ? "completed" : "partial");
  1435→    return ctx;
  1436→  } catch (e: any) {
  1437→    log(ctx, "error", `خطای بحرانی در اجرای ایجنت: ${e.message}`);
  1438→    ctx.errors.push(`CRITICAL: ${e.message}`);
  1439→    await finishRun(ctx, "failed");
  1440→    return ctx;
  1441→  }
  1442→}

  1461→async function finishRun(ctx: RunContext, status: string) {
  1462→  const durationMs = Date.now() - ctx.startedAt;
  1463→  await db.seoAgentRun
  1464→    .update({
  1465→      where: { id: ctx.runId },
  1466→      data: {
  1467→        status,
  1468→        successCount: ctx.successCount,
  1469→        failCount: ctx.failCount,
  1470→        logs: JSON.stringify(ctx.logs),
  1471→        results: JSON.stringify({
  1472→          articles: ctx.articles,
  1473→          errors: ctx.errors,
  1474→        }),
  1475→        durationMs,
  1476→        finishedAt: new Date(),
  1477→      },

  1484→      data: { lastRunAt: new Date() },

  1504→export async function startBackgroundRun(opts: {
  1505→  adminId: string;
  1506→  mode: "full" | "continue" | "strategy_only";
  1507→  count: number;
  1508→  publishImmediately?: boolean;
  1509→}): Promise<{ runId: string; alreadyRunning: boolean }> {
  1510→  if (isAgentRunning()) {
  1511→    return { runId: getCurrentRunId() || "", alreadyRunning: true };
  1512→  }
  1513→  // Create run record
  1514→  const run = await db.seoAgentRun.create({
  1515→    data: {
  1516→      mode: opts.mode,
  1517→      requestedCount: opts.count,
  1518→      status: "running",
  1519→      startedAt: new Date(),
  1520→    },
  1521→  });
  1522→  globalThis.__seoAgentRunning = true;
  1523→  globalThis.__seoAgentRunId = run.id;
  1524→
  1525→  // Fire and forget — run in background
  1526→  runSeoAgent({
  1527→    runId: run.id,
  1528→    adminId: opts.adminId,
  1529→    mode: opts.mode,
  1530→    count: opts.count,
  1531→    publishImmediately: opts.publishImmediately,
  1532→  })
  1533→    .catch((e) => {
  1534→      console.error("[SEO-AGENT] background run crashed:", e);
  1535→    })
  1536→    .finally(() => {
  1537→      globalThis.__seoAgentRunning = false;
  1538→      globalThis.__seoAgentRunId = undefined;
  1539→    });
  1540→
  1541→  return { runId: run.id, alreadyRunning: false };
  1542→}
