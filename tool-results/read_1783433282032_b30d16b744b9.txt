   296→export function AdminOverlay({ standalone = false }: { standalone?: boolean } = {}) {
   297→  const { setOverlay, reset } = useAppStore();
   298→  const [tab, setTab] = useState<AdminTab>("dashboard");
   299→  const [showSettings, setShowSettings] = useState(false);
   300→  const [permissions, setPermissions] = useState<AdminPermissions>(ALL_TRUE_PERMISSIONS);
   301→  const [permsLoaded, setPermsLoaded] = useState(false);
   302→
   303→  // Fetch the current admin's permissions on mount — used to filter visible tabs
   304→  useEffect(() => {
   305→    (async () => {
   306→      try {
   307→        const res = await fetch("/api/admin/permissions", { cache: "no-store" });
   308→        if (res.ok) {
   309→          const data = await res.json();
   310→          if (data?.permissions) setPermissions(data.permissions as AdminPermissions);
   311→        }
   312→      } catch {
   313→        // fall back to all-true if the call fails (best-effort)
   314→      } finally {
   315→        setPermsLoaded(true);
   316→      }
   317→    })();
   318→  }, []);
   319→
   320→  async function handleLogout() {
   321→    try {
   322→      await fetch("/api/auth/logout", { method: "POST" });
   323→      reset();
   324→      toast.success("از پنل مدیریت خارج شدید");
   325→    } catch {
   326→      toast.error("خطا در خروج");
   327→    }
   328→  }
   329→
   330→  // All possible tabs — we filter the visible ones based on permissions
   331→  const allTabs: { id: AdminTab; label: string; icon: any; perm: PermissionKey }[] = [
   332→    { id: "dashboard", label: "داشبورد", icon: LayoutDashboard, perm: "canViewDashboard" },

   336→    { id: "programs", label: "صف برنامه‌ها", icon: ListChecks, perm: "canManagePrograms" },

   338→    { id: "articles", label: "مقالات", icon: Newspaper, perm: "canManageArticles" },
   339→    { id: "head_codes", label: "کدهای تحلیلی", icon: Code2, perm: "canManageHeadCodes" },
   340→    { id: "terms", label: "قوانین", icon: ShieldCheck, perm: "canManageTerms" },
   341→    { id: "tickets", label: "تیکت‌ها", icon: MessageSquare, perm: "canManageTickets" },

   786→function ChargeWalletDialog({ user, onClose, onDone }: { user: any; onClose: () => void; onDone: () => void }) {
   787→  const [amount, setAmount] = useState("");
   788→  const [desc, setDesc] = useState("");
   789→  const [saving, setSaving] = useState(false);
   790→
   791→  async function charge() {
   792→    if (!amount || Number(amount) <= 0) return;
   793→    setSaving(true);
   794→    try {
   795→      const res = await fetch("/api/admin/wallet-charge", {
   796→        method: "POST", headers: { "Content-Type": "application/json" },
   797→        body: JSON.stringify({ userId: user.id, amount: Number(amount), description: desc }),
   798→      });
   799→      if (!res.ok) throw new Error();
   800→      toast.success(`کیف پول ${toPersianDigits(formatToman(Number(amount)))} ت شارژ شد`);
   801→      onDone();
   802→    } catch { toast.error("خطا در شارژ"); } finally { setSaving(false); }
   803→  }
   804→
   805→  return (
   806→    <Dialog open onOpenChange={onClose}>
   807→      <DialogContent dir="rtl" className="max-w-sm">
   808→        <DialogHeader><DialogTitle>شارژ کیف پول — {user.name || user.mobile}</DialogTitle></DialogHeader>
   809→        <div className="space-y-3">
   810→          <div className="p-2 rounded-lg bg-muted/40 text-xs text-center">موجودی فعلی: <b className="font-stat">{toPersianDigits(formatToman(user.walletBalance || 0))} ت</b></div>
   811→          <div>
   812→            <Label className="mb-1 block">مبلغ (تومان)</Label>
   813→            <Input type="number" dir="ltr" value={amount} onChange={(e) => setAmount(e.target.value)} placeholder="مثلاً 500000" className="rounded-xl text-center font-stat" />
   814→          </div>
   815→          <div>
   816→            <Label className="mb-1 block">توضیحات (اختیاری)</Label>
   817→            <Input value={desc} onChange={(e) => setDesc(e.target.value)} placeholder="شارژ تستی / پشتیبانی" className="rounded-xl" />
   818→          </div>
