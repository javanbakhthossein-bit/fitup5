    90→// Build user context string from onboarding + plan tier
    91→export function buildUserContext(data: OnboardingData, planName?: Plan | null): string {
    92→  const bmr =
    93→    data.gender === "male"
    94→      ? 10 * data.weight + 6.25 * data.height - 5 * data.age + 5
    95→      : 10 * data.weight + 6.25 * data.height - 5 * data.age - 161;
    96→  const caps = getCapabilities(planName ?? null);
    97→  const planLabel = planName ? PLAN_LABELS[planName] : "بدون پلن (Basic)";
    98→  return `اطلاعات کاربر:
    99→- جنسیت: ${GENDER_LABELS[data.gender]}
   100→- سن: ${data.age} سال
   101→- قد: ${data.height} سانتی‌متر
   102→- وزن: ${data.weight} کیلوگرم
   103→- وزن هدف: ${data.targetWeight ?? "نامشخص"} کیلوگرم
   104→- هدف اصلی: ${GOAL_LABELS[data.goal]}
   105→- سطح فعالیت: ${ACTIVITY_LABELS[data.activityLevel]}
   106→- روزهای تمرین در هفته: ${data.workoutDays} روز${data.workoutDaysList && data.workoutDaysList.length > 0 ? ` (${data.workoutDaysList.join("، ")})` : ""}
   107→- مکان تمرین: ${WORKOUT_PLACE_LABELS[data.workoutPlace]}
   108→- تجهیزات: ${data.equipment.length ? data.equipment.join("، ") : "بدون تجهیزات خاص"}
   109→- سوابق بیماری: ${data.diseases || "ندارد"}
   110→- آسیب‌دیدگی مفصلی/عضلانی: ${data.injuries || "ندارد"}
   111→- حساسیت غذایی: ${data.allergies || "ندارد"}
   112→- نوع رژیم غذایی: ${DIET_LABELS[data.dietType]}
   113→${data.trainingExperience ? `- سابقه ورزشی: ${TRAINING_EXPERIENCE_LABELS[data.trainingExperience]}` : ""}
   114→${data.previousTrainingType ? `- نوع تمرین قبلی: ${data.previousTrainingType}` : ""}
   115→${data.drugAllergies ? `- آلرژی‌های دارویی: ${data.drugAllergies}` : ""}
   116→${data.currentMedications ? `- داروهای مصرفی: ${data.currentMedications}` : ""}
   117→${data.maxLifts ? `- حداکثر وزنه‌ها: ${data.maxLifts}` : ""}
   118→- BMR تخمینی: ${Math.round(bmr)} کالری
   119→- پلن اشتراک کاربر: ${planLabel}
   120→- قابلیت‌های فعال: ${[
   121→    caps.workoutAndNutritionPlan ? "برنامه تمرین+تغذیه" : null,
   122→    caps.supplementsPlan ? "برنامه مکمل" : null,
   123→    caps.periodicCheckups ? "چکاپ دوره‌ای" : null,
   124→    caps.aiChatQuestions !== 0 ? "چت هوشمند" : null,
   125→    caps.mealPhotoAnalysis ? "آنالیز عکس غذا" : null,

   133→/**
   134→ * ساخت بلوک پرامپت مخصوص پلن کاربر.
   135→ * بر اساس سطح پلن، سطح جزئیات و دامنه برنامه را تعیین می‌کند.
   136→ * - Basic: فقط برنامه تمرین + تغذیه
   137→ * - Standard+: اضافه‌کردن برنامه مکمل‌ها
   138→ * - Ultimate: تزریق متغیرهای آزمایش خون و آنالیز ویدیویی
   139→ */
   140→export function buildPlanAwareInstructions(planName?: Plan | null, extras?: { bloodTestReport?: string; videoAnalysisResult?: string }): string {
   141→  const caps = getCapabilities(planName ?? null);
   142→  const instructions: string[] = [];
   143→
   144→  if (caps.supplementsPlan) {
   145→    instructions.push(
   146→      "برنامه مکمل‌های ورزشی: یک بخش «supplements» در JSON اضافه کن شامل مکمل‌های پیشنهادی با دوز ایمن و علمی. فقط مکمل‌های زیر مجاز هستند:\n" +
   147→      "• کراتین مونوهیدرات: ۳-۵ گرم در روز (هر روز)\n" +
   148→      "• پروتئین وی: ۲۵-۳۰ گرم در وعده (بعد تمرین)\n" +
   149→      "• امگا ۳ (روغن ماهی): ۱-۲ گرم در روز\n" +
   150→      "• مولتی‌ویتامین: ۱ قرص در روز\n" +
   151→      "• ویتامین D3: ۱۰۰۰-۲۰۰۰ واحد در روز (نه بیشتر!)\n" +
   152→      "• ال-گلوتامین: ۵ گرم در روز\n" +

   154→    );
   155→  }
   156→
   157→  if (caps.videoBodyAnalysis && extras?.videoAnalysisResult) {
   158→    instructions.push(
   159→      `آنالیز ویدیویی بدن کاربر (توسط هوش مصنوعی بررسی شده):\n${extras.videoAnalysisResult}\nاین اطلاعات را در طراحی برنامه لحاظ کن — نقاط ضعف فرم بدن، عدم تقارن، و حرکات اصلاحی پیشنهاد بده.`
   160→    );
   161→  }
   162→
   163→  if (caps.bloodTestAnalysis && extras?.bloodTestReport) {
   164→    instructions.push(
   165→      `گزارش آزمایش خون کاربر:\n${extras.bloodTestReport}\nبرنامه تغذیه را بر اساس کمبودهای ویتامینی، سطح قند، چربی خون و هورمون‌ها بهینه‌سازی کن. در صورت کمبود، غذاهای غنی از آن ویتامین/ماده معدنی را اولویت بده. هشدارهای پزشکی لازم را در بخش notes بیاور.`
   166→    );
   167→  }
   168→
   169→  return instructions.length
   170→    ? `\n\nدستورالعمل‌های ویژه پلن ${planName ? PLAN_LABELS[planName] : ""}:\n${instructions.map((i, idx) => `${idx + 1}. ${i}`).join("\n")}`
   171→    : "";
   172→}

   184→  // Use user-selected specific weekdays if provided; otherwise fallback to first N days

   186→    data.workoutDaysList && data.workoutDaysList.length > 0
   187→      ? data.workoutDaysList
   188→      : PERSIAN_WEEKDAYS.slice(0, data.workoutDays);
   189→
   190→  // --- Fetch exercise names from the library so AI only uses real exercises ---
   191→  let libraryNames: string[] = [];

   201→
   202→  const userPrompt = `بر اساس اطلاعات زیر، یک برنامه تمرینی هفتگی کامل و شخصی‌سازی‌شده بساز.
   203→
   204→${context}
   205→${planInstructions}

   208→روزهای تمرین کاربر: ${chosenDays.join("، ")}
   209→
   210→فقط و فقط با ساختار JSON زیر پاسخ بده و هیچ متن اضافه‌ای قبل یا بعد از JSON ننویس:

   213→    {
   214→      "day": "شنبه",
   215→      "title": "عنوان روز تمرین",

   217→      "estimatedMinutes": 60,
   218→      "exercises": [
   219→        {
   220→          "name": "نام حرکت (حتماً از کتابخانه حرکات بالا)",
   221→          "muscle": "عضله هدف",
   222→          "category": "push|pull|legs|core|cardio|fullbody",
   223→          "description": "توضیح نحوه انجام (۲-۳ جمله کامل و واضح)",

   225→          "difficulty": "beginner|intermediate|advanced",
   226→          "sets": [
   227→            {"setNumber": 1, "reps": "10-12", "restSec": 90},
   228→            {"setNumber": 2, "reps": "10-12", "restSec": 90}
   229→          ],
   230→          "supersetGroup": "A",
   231→          "supersetType": "superset"
   232→        }
   233→      ]
   234→    }
   235→  ],
   236→  "weeklyGoal": "هدف هفته",
   237→  "notes": "نکات کلی هفته"
   238→}
   239→
   240→قوانین مهم:
   241→- دقیقاً ${data.workoutDays} روز تمرین در روزهای ذکر شده (${chosenDays.join("، ")}) ایجاد کن.

   269→  const enriched: WorkoutPlanContent = {
   270→    days: (parsed.days || []).map((day: any) => ({
   271→      ...day,
   272→      exercises: (day.exercises || []).map((ex: any, i: number) => ({

   281→        })),
   282→        supersetGroup: ex.supersetGroup || undefined,
   283→        supersetType: ex.supersetType || undefined,

   285→    })),

   883→// Parse JSON from LLM content (handles markdown code fences + HTML error pages gracefully)
   884→function parseJsonFromContent(content: string): any {
   885→  if (!content || typeof content !== "string") {
   886→    return { days: [], meals: [], notes: "پاسخ هوش مصنوعی خالی بود." };
   887→  }
   888→
   889→  let cleaned = content.trim();
   890→
   891→  // Detect HTML error pages (gateway error, 502/504, AvalAI returning HTML)
   892→  // If the content looks like HTML and contains no JSON braces, bail out gracefully.
   893→  const looksLikeHtml = /^\s*<(?:html|!doctype|head|body|h1|div|p)\b/i.test(cleaned) ||
   894→    (cleaned.startsWith("<") && cleaned.includes("</") && !cleaned.includes("{"));
   895→  if (looksLikeHtml) {
   896→    console.error("[parseJsonFromContent] AI returned HTML (likely an error page):", cleaned.slice(0, 200));
   897→    return { days: [], meals: [], notes: "خطا در پاسخ هوش مصنوعی. لطفاً دوباره تلاش کنید." };
   898→  }
   899→
   900→  // Extract JSON from ```json ... ``` or ``` ... ```
   901→  const fenceMatch = cleaned.match(/```(?:json)?\s*([\s\S]*?)```/);
   902→  if (fenceMatch) {
   903→    cleaned = fenceMatch[1].trim();
   904→  }
   905→
   906→  // Find first { and last }
   907→  const first = cleaned.indexOf("{");
   908→  const last = cleaned.lastIndexOf("}");
   909→  if (first !== -1 && last !== -1 && last > first) {
   910→    cleaned = cleaned.slice(first, last + 1);
   911→  }
   912→
   913→  try {
   914→    return JSON.parse(cleaned);
   915→  } catch (err) {
   916→    console.error("[parseJsonFromContent] JSON parse failed. Raw content:", content.slice(0, 300), err);
   917→    return { days: [], meals: [], notes: "خطا در پردازش پاسخ هوش مصنوعی." };
   918→  }
   919→}
