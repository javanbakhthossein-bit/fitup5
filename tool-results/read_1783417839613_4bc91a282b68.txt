   294→export function AdminOverlay({ standalone = false }: { standalone?: boolean } = {}) {
   295→  const { setOverlay, reset } = useAppStore();
   296→  const [tab, setTab] = useState<AdminTab>("dashboard");
   297→  const [showSettings, setShowSettings] = useState(false);
   298→  const [permissions, setPermissions] = useState<AdminPermissions>(ALL_TRUE_PERMISSIONS);
   299→  const [permsLoaded, setPermsLoaded] = useState(false);
   300→
   301→  // Fetch the current admin's permissions on mount — used to filter visible tabs
   302→  useEffect(() => {
   303→    (async () => {
   304→      try {
   305→        const res = await fetch("/api/admin/permissions", { cache: "no-store" });
   306→        if (res.ok) {
   307→          const data = await res.json();
   308→          if (data?.permissions) setPermissions(data.permissions as AdminPermissions);
   309→        }
   310→      } catch {
   311→        // fall back to all-true if the call fails (best-effort)
   312→      } finally {
   313→        setPermsLoaded(true);
   314→      }
   315→    })();
   316→  }, []);
   317→
   318→  async function handleLogout() {
   319→    try {
   320→      await fetch("/api/auth/logout", { method: "POST" });
   321→      reset();
   322→      toast.success("از پنل مدیریت خارج شدید");
   323→    } catch {
   324→      toast.error("خطا در خروج");
   325→    }
   326→  }
   327→
   328→  // All possible tabs — we filter the visible ones based on permissions
   329→  const allTabs: { id: AdminTab; label: string; icon: any; perm: PermissionKey }[] = [
   330→    { id: "dashboard", label: "داشبورد", icon: LayoutDashboard, perm: "canViewDashboard" },
   331→    { id: "users", label: "کاربران", icon: Users, perm: "canManageUsers" },
   332→    { id: "finance", label: "مالی و تراکنش‌ها", icon: Wallet, perm: "canViewFinance" },
   333→    { id: "programs", label: "صف برنامه‌ها", icon: ListChecks, perm: "canManagePrograms" },
   334→    { id: "checkups", label: "چکاپ‌ها", icon: ClipboardCheck, perm: "canManageCheckups" },
   335→    { id: "articles", label: "مقالات", icon: Newspaper, perm: "canManageArticles" },
   336→    { id: "head_codes", label: "کدهای تحلیلی", icon: Code2, perm: "canManageHeadCodes" },
   337→    { id: "terms", label: "قوانین", icon: ShieldCheck, perm: "canManageTerms" },
   338→    { id: "tickets", label: "تیکت‌ها", icon: MessageSquare, perm: "canManageTickets" },
   339→    { id: "copilot", label: "دستیار هوشمند", icon: Sparkles, perm: "canUseCopilot" },

   341→    { id: "domain", label: "دامنه و رکوردها", icon: Globe, perm: "canManageHeadCodes" },

   343→    { id: "logs", label: "لاگ خطاها", icon: Bug, perm: "canViewDashboard" },
   344→  ];
   345→
   346→  // Filter tabs by permission; always show at least the dashboard
   347→  const tabs = allTabs.filter((t) => permissions[t.perm]);
   348→
   349→  // If the current tab is not visible (e.g. perms changed), fall back to first available
   350→  useEffect(() => {
   351→    if (permsLoaded && tabs.length > 0 && !tabs.some((t) => t.id === tab)) {
   352→      setTab(tabs[0].id);
   353→    }
   354→  }, [permsLoaded, permissions, tabs, tab]);
   355→
   356→  return (

   371→              <LogOut className="w-4 h-4" />
   372→              <span className="hidden sm:inline">خروج</span>

   409→            {tab === "terms" && <TermsTab />}
   410→            {tab === "tickets" && <TicketsTab />}

   420→      {/* Site Settings Dialog */}

   778→function ChargeWalletDialog({ user, onClose, onDone }: { user: any; onClose: () => void; onDone: () => void }) {
   779→  const [amount, setAmount] = useState("");
   780→  const [desc, setDesc] = useState("");
   781→  const [saving, setSaving] = useState(false);
   782→
   783→  async function charge() {
   784→    if (!amount || Number(amount) <= 0) return;
   785→    setSaving(true);
   786→    try {
   787→      const res = await fetch("/api/admin/wallet-charge", {
   788→        method: "POST", headers: { "Content-Type": "application/json" },
   789→        body: JSON.stringify({ userId: user.id, amount: Number(amount), description: desc }),
   790→      });
   791→      if (!res.ok) throw new Error();
   792→      toast.success(`کیف پول ${toPersianDigits(formatToman(Number(amount)))} ت شارژ شد`);
   793→      onDone();
   794→    } catch { toast.error("خطا در شارژ"); } finally { setSaving(false); }
   795→  }
   796→
   797→  return (
   798→    <Dialog open onOpenChange={onClose}>
   799→      <DialogContent dir="rtl" className="max-w-sm">
   800→        <DialogHeader><DialogTitle>شارژ کیف پول — {user.name || user.mobile}</DialogTitle></DialogHeader>
   801→        <div className="space-y-3">
   802→          <div className="p-2 rounded-lg bg-muted/40 text-xs text-center">موجودی فعلی: <b className="font-stat">{toPersianDigits(formatToman(user.walletBalance || 0))} ت</b></div>
   803→          <div>
   804→            <Label className="mb-1 block">مبلغ (تومان)</Label>

   807→          <div>

   810→          </div>
   811→          <div className="flex gap-2 flex-wrap">
   812→            {[100000, 500000, 1000000].map(v => (
   813→              <button key={v} onClick={() => setAmount(String(v))} className="text-xs px-3 py-1.5 rounded-lg bg-muted hover:bg-primary hover:text-primary-foreground transition font-stat">

   815→              </button>
   816→            ))}

   820→          <Button onClick={onClose} variant="outline" className="rounded-xl">انصراف</Button>

   825→  );
