   802→   802→// Parse JSON from LLM content (handles markdown code fences)
   803→   803→function parseJsonFromContent(content: string): any {
   804→   804→  // Remove markdown code fences if present
   805→   805→  let cleaned = content.trim();
   806→   806→  // Extract JSON from ```json ... ``` or ``` ... ```
   807→   807→  const fenceMatch = cleaned.match(/```(?:json)?\s*([\s\S]*?)```/);
   808→   808→  if (fenceMatch) {
   809→   809→    cleaned = fenceMatch[1].trim();
   810→   810→  }
   811→   811→  // Find first { and last }
   812→   812→  const first = cleaned.indexOf("{");
   813→   813→  const last = cleaned.lastIndexOf("}");
   814→   814→  if (first !== -1 && last !== -1 && last > first) {
   815→   815→    cleaned = cleaned.slice(first, last + 1);
   816→   816→  }
   817→   817→  try {
   818→   818→    return JSON.parse(cleaned);
   819→   819→  } catch {
   820→   820→    return { days: [], meals: [], notes: "خطا در پردازش پاسخ هوش مصنوعی." };
   821→   821→  }
