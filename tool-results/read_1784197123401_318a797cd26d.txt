    76→const videoUrlCache = new Map<string, string>();
    77→
    78→export async function fetchExerciseVideoUrl(name: string): Promise<string> {
    79→  if (videoUrlCache.has(name)) return videoUrlCache.get(name)!;
    80→  try {
    81→    const res = await fetch(`/api/exercises?search=${encodeURIComponent(name)}`);
    82→    if (!res.ok) return "";
    83→    const data = await res.json();
    84→    const exercises: any[] = data.exercises || [];
    85→    // تطبیق دقیق نام، در غیر این صورت اولین نتیجه
    86→    const match = exercises.find((e) => e.name === name) || exercises[0];
    87→    const url = match?.youtubeUrl || "";
    88→    videoUrlCache.set(name, url);
    89→    return url;
    90→  } catch {
    91→    return "";
    92→  }
    93→}
