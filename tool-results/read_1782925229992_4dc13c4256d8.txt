    88→// Build user context string from onboarding + plan tier
    89→export function buildUserContext(data: OnboardingData, planName?: Plan | null): string {
    90→  const bmr =
    91→    data.gender === "male"
    92→      ? 10 * data.weight + 6.25 * data.height - 5 * data.age + 5
    93→      : 10 * data.weight + 6.25 * data.height - 5 * data.age - 161;
    94→  const caps = getCapabilities(planName ?? null);
    95→  const planLabel = planName ? PLAN_LABELS[planName] : "بدون پلن (Basic)";
    96→  return `اطلاعات کاربر:
    97→- جنسیت: ${GENDER_LABELS[data.gender]}
    98→- سن: ${data.age} سال
    99→- قد: ${data.height} سانتی‌متر
   100→- وزن: ${data.weight} کیلوگرم
   101→- وزن هدف: ${data.targetWeight ?? "نامشخص"} کیلوگرم
   102→- هدف اصلی: ${GOAL_LABELS[data.goal]}
   103→- سطح فعالیت: ${ACTIVITY_LABELS[data.activityLevel]}
   104→- روزهای تمرین در هفته: ${data.workoutDays} روز
   105→- مکان تمرین: ${WORKOUT_PLACE_LABELS[data.workoutPlace]}

   108→- آسیب‌دیدگی مفصلی/عضلانی: ${data.injuries || "ندارد"}

   126→/**

   199→        }
   200→      ]
   201→    }
   202→  ],
   203→  "weeklyGoal": "هدف هفته",
   204→  "notes": "نکات کلی هفته"
   205→}

   208→- دقیقاً ${data.workoutDays} روز تمرین در روزهای ذکر شده ایجاد کن.

   250→  };
   251→
   252→  return enriched;
   253→}

   256→export async function generateMealPlan(
   257→  data: OnboardingData,
   258→  planName?: Plan | null,
   259→  extras?: { bloodTestReport?: string }
   260→): Promise<MealPlanContent> {
   261→  const systemPrompt = await getAiConfig(
   262→    "nutrition_system_prompt",
   263→    DEFAULT_NUTRITION_PROMPT
   264→  );
   265→  const context = buildUserContext(data, planName);
   266→  const planInstructions = buildPlanAwareInstructions(planName, extras);
   267→
   268→  // Calculate target calories based on goal
   269→  const bmr =
   270→    data.gender === "male"
   271→      ? 10 * data.weight + 6.25 * data.height - 5 * data.age + 5
   272→      : 10 * data.weight + 6.25 * data.height - 5 * data.age - 161;
   273→  const tdee = bmr * 1.4;
   274→  let targetCal = tdee;
   275→  if (data.goal === "fat_loss") targetCal = tdee - 400;

   802→// Parse JSON from LLM content (handles markdown code fences)
   803→function parseJsonFromContent(content: string): any {
   804→  // Remove markdown code fences if present
   805→  let cleaned = content.trim();
   806→  // Extract JSON from ```json ... ``` or ``` ... ```
   807→  const fenceMatch = cleaned.match(/```(?:json)?\s*([\s\S]*?)```/);
   808→  if (fenceMatch) {
   809→    cleaned = fenceMatch[1].trim();
   810→  }
   811→  // Find first { and last }
   812→  const first = cleaned.indexOf("{");
   813→  const last = cleaned.lastIndexOf("}");
   814→  if (first !== -1 && last !== -1 && last > first) {
   815→    cleaned = cleaned.slice(first, last + 1);
   816→  }
   817→  try {
   818→    return JSON.parse(cleaned);
   819→  } catch {
   820→    return { days: [], meals: [], notes: "خطا در پردازش پاسخ هوش مصنوعی." };
   821→  }
   822→}
