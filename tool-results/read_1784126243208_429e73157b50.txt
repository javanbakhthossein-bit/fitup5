   100→function log(
   101→  ctx: RunContext,
   102→  level: RunProgress["level"],
   103→  msg: string
   104→): void {
   105→  const entry: RunProgress = { level, msg, ts: Date.now() };
   106→  ctx.logs.push(entry);
   107→  // Also print to server log for debugging
   108→  const prefix =
   109→    level === "error"
   110→      ? "[SEO-AGENT:ERROR]"
   111→      : level === "warn"
   112→      ? "[SEO-AGENT:WARN]"
   113→      : level === "success"
   114→      ? "[SEO-AGENT:OK]"
   115→      : "[SEO-AGENT]";
   116→  console.log(`${prefix} ${msg}`);
   117→  // Persist to DB every 5 logs or every 3 seconds — for live UI updates
   118→  const now = Date.now();
   119→  const lastPersist = (ctx as any)._lastPersist || 0;
   120→  if (ctx.logs.length % 5 === 0 || now - lastPersist > 3000) {
   121→    (ctx as any)._lastPersist = now;
   122→    // fire-and-forget — don't block the agent
   123→    db.seoAgentRun
   124→      .update({
   125→        where: { id: ctx.runId },
   126→        data: {
   127→          logs: JSON.stringify(ctx.logs.slice(-200)),
   128→          successCount: ctx.successCount,
   129→          failCount: ctx.failCount,
   130→        },
   131→      })
   132→      .catch(() => null);
   133→  }
   134→}

   136→// ─── JSON extraction helper ───
   137→function extractJson(content: string): any | null {
   138→  if (!content) return null;
   139→  // Try code block first
   140→  const codeBlock = content.match(/```(?:json)?\s*([\s\S]*?)```/);
   141→  const candidate = codeBlock ? codeBlock[1].trim() : content.trim();
   142→  // Find the first balanced { ... }
   143→  const start = candidate.indexOf("{");
   144→  if (start === -1) return null;
   145→  let depth = 0;
   146→  for (let i = start; i < candidate.length; i++) {
   147→    if (candidate[i] === "{") depth++;
   148→    else if (candidate[i] === "}") {
   149→      depth--;
   150→      if (depth === 0) {
   151→        const jsonStr = candidate.slice(start, i + 1);
   152→        try {
   153→          return JSON.parse(jsonStr);
   154→        } catch {
   155→          // try to fix common issues
   156→          try {
   157→            return JSON.parse(
   158→              jsonStr
   159→                .replace(/,\s*}/g, "}")
   160→                .replace(/,\s*]/g, "]")
   161→                .replace(/[\u201C\u201D]/g, '"')
   162→                .replace(/[\u2018\u2019]/g, "'")
   163→            );
   164→          } catch {
   165→            return null;
   166→          }
   167→        }
   168→      }
   169→    }
   170→  }
   171→  return null;
   172→}

   174→async function callLlm(system: string, user: string, maxTokens = 4000): Promise<string> {
   175→  // timeout ۳ دقیقه — برای production که AI کندتر پاسخ می‌دهد
   176→  const timeoutMs = 180000;
   177→  const controller = new AbortController();
   178→  const timeout = setTimeout(() => controller.abort(), timeoutMs);
   179→  try {
   180→    const completion = await getAvalaiClient().chat.completions.create({
   181→      model: TEXT_MODEL,
   182→      messages: [
   183→        { role: "system", content: system },
   184→        { role: "user", content: user },
   185→      ],
   186→      max_tokens: maxTokens,
   187→      temperature: 0.7,
   188→      // @ts-expect-error — OpenAI SDK supports signal
   189→      signal: controller.signal,
   190→    } as any);
   191→    return completion.choices[0]?.message?.content || "";
   192→  } catch (e: any) {
   193→    if (e.name === "AbortError") {
   194→      throw new Error(`AI timeout after ${timeoutMs / 1000}s`);
   195→    }
   196→    throw e;
   197→  } finally {
   198→    clearTimeout(timeout);
   199→  }
   200→}

   203→export async function analyzeSite(ctx: RunContext): Promise<SiteAnalysis> {
   204→  log(ctx, "info", "🔍 شروع تحلیل سایت — استخراج صفحات و مقالات موجود");
   205→  // Static landing pages & tools (همیشه در سایت فیتاپ وجود دارند)
   206→  // مهم: همه مسیرهای معتبر لینک داخلی در سایت فیتاپ. از این لیست برای لینک‌سازی استفاده کن.
   207→  const pages = [
   208→    { url: "/", title: "فیتاپ — مربی هوشمند بدنسازی و تغذیه", type: "landing" as const },
   209→    { url: "/?screen=articles", title: "مقالات ورزشی و تغذیه فیتاپ", type: "list" as const },
   210→    { url: "/?tool=tdee", title: "محاسبه‌گر کالری و TDEE", type: "tool" as const },
   211→    { url: "/?tool=exercises", title: "بانک حرکات ورزشی", type: "tool" as const },
   212→    { url: "/?tool=foods", title: "جدول کالری غذاها", type: "tool" as const },
   213→    { url: "/?screen=auth", title: "ورود / ثبت‌نام فیتاپ", type: "landing" as const },
   214→  ];
   215→
   216→  const articles = await db.article.findMany({
   217→    where: { status: "published" },
   218→    select: {
   219→      id: true,
   220→      title: true,
   221→      slug: true,
   222→      category: true,
   223→      tags: true,
   224→      excerpt: true,
   225→    },
   226→    orderBy: { createdAt: "desc" },
   227→  });
   228→
   229→  // Collect existing keywords from tags
   230→  const existingKeywords = new Set<string>();
   231→  for (const a of articles) {
   232→    if (a.tags) {
   233→      for (const t of a.tags.split(",").map((s) => s.trim()).filter(Boolean)) {
   234→        existingKeywords.add(t);
   235→      }
   236→    }
   237→  }
   238→
   239→  const totalViewsAgg = await db.article.aggregate({
   240→    where: { status: "published" },
   241→    _sum: { views: true },
   242→  });
   243→
   244→  const result: SiteAnalysis = {
   245→    pages,
   246→    articles: articles.map((a) => ({
   247→      id: a.id,
   248→      title: a.title,
   249→      slug: a.slug,
   250→      category: a.category,
   251→      tags: a.tags,
   252→      excerpt: a.excerpt,
   253→    })),
   254→    existingKeywords: Array.from(existingKeywords),
   255→    totalArticles: articles.length,
   256→    totalViews: totalViewsAgg._sum.views || 0,
   257→  };
   258→
   259→  log(
   260→    ctx,
   261→    "success",
   262→    `✅ تحلیل سایت کامل شد: ${result.pages.length} صفحه کلیدی، ${result.totalArticles} مقاله موجود، ${result.existingKeywords.length} کلمه کلیدی موجود، ${result.totalViews} بازدید کل`
   263→  );
   264→  return result;
   265→}

   625→/**
   626→ * Compute the scheduled publish date for a new article.
   627→ *
   628→ * Scheduling rules:
   629→ *  1. If the strategy's contentCalendar has an entry whose keyword matches
   630→ *     (case-insensitive substring) the article's plan keyword, the article is
   631→ *     scheduled for `now + (week * 7 days)`. Week 1 = next week, week 2 = 2
   632→ *     weeks from now, etc. — this spreads articles across the calendar as the
   633→ *     strategy intends.
   634→ *  2. Otherwise, the article is scheduled 7 days after the latest already-scheduled
   635→ *     draft article (so articles never collide on the same day). If no draft is
   636→ *     currently scheduled, it's scheduled 1 day from now as a sensible default.
   637→ *  3. The returned date is always strictly in the future (>= now + 1 hour) so the
   638→ *     cron publisher won't immediately fire on the same run.
   639→ *
   640→ * Returns null when `publishImmediately` is true (caller should publish now).
   641→ */
   642→export async function computeScheduledAt(
   643→  plan: ArticlePlan,
   644→  strategy: SeoStrategyContent | null | undefined,
   645→  publishImmediately: boolean = false
   646→): Promise<Date | null> {
   647→  if (publishImmediately) return null;
   648→
   649→  const now = new Date();
   650→  const oneHourFromNow = new Date(now.getTime() + 60 * 60 * 1000);
   651→
   652→  // 1) پیدا کردن آخرین مقاله زمان‌بندی‌شده (draft با scheduledAt)
   653→  const latestScheduled = await db.article.findFirst({
   654→    where: {
   655→      status: "draft",
   656→      scheduledAt: { not: null },
   657→    },
   658→    orderBy: { scheduledAt: "desc" },
   659→    select: { scheduledAt: true },
   660→  });
   661→
   662→  // 2) محاسبه تاریخ پایه
   663→  // اگر مقاله زمان‌بندی‌شده‌ای وجود دارد، از آن استفاده کن
   664→  // در غیر این صورت، از امروز استفاده کن
   665→  let baseDate = now;
   666→  if (latestScheduled?.scheduledAt) {
   667→    baseDate = new Date(latestScheduled.scheduledAt);
   668→  }
   669→
   670→  // 3) فاصله دقیقاً ۴ روز بین هر مقاله
   671→  // این تضمین می‌کند که مقالات با فاصله منظم ۴ روز منتشر شوند
   672→  // مثلاً ۱۰ مقاله = ۴۰ روز انتشار
   673→  const gapDays = 4;
   674→
   675→  const target = new Date(baseDate);
   676→  target.setDate(target.getDate() + gapDays);
   677→  target.setHours(9, 0, 0, 0); // ساعت ۹ صبح
   678→
   679→  // 4) مطمئن شو که تاریخ در آینده است
   680→  if (target.getTime() < oneHourFromNow.getTime()) {
   681→    // اگر تاریخ محاسبه‌شده در گذشته است، ۲ روز از امروز استفاده کن
   682→    const fallback = new Date(now);
   683→    fallback.setDate(fallback.getDate() + 2);
   684→    fallback.setHours(9, 0, 0, 0);
   685→    return fallback;
   686→  }
   687→
   688→  return target;
   689→}

   763→لینک‌های داخلی پیشنهادی (حتماً در متن از این‌ها استفاده کن):
   764→${plan.internalLinks
   765→  .map((l) => `- [${l.anchor}](/?article=${l.slug}) — هدف: "${l.title}"`)
   766→  .join("\n")}
   767→
   768→سایر صفحات/مقالات موجود برای لینک‌سازی اضافی (به این‌ها هم با فرمت /?article=slug یا /?tool=... لینک بزن):
   769→${existingForLinks
   770→  .map((p) => `- slug="${p.slug}" title="${p.title}"`)
   771→  .join("\n")}
   772→
   773→یادآوری فرمت لینک‌ها:
   774→- لینک به ابزار: [متن](/?tool=tdee) یا [متن](/?tool=exercises) یا [متن](/?tool=foods)
   775→- لینک به مقاله: [متن](/?article=slug)
   776→- لینک به صفحه اصلی: [متن](/)
   777→- لینک به لیست مقالات: [متن](/?screen=articles)
   778→
   779→فقط محتوای Markdown بنویس. بدون توضیح اضافه در ابتدا یا انتها.`;
   780→
   781→  let articleContent = "";
   782→  try {
   783→    articleContent = await callLlm(contentSystem, contentUser, 8000);
   784→  } catch (e: any) {
   785→    throw new Error(`خطا در تولید محتوا: ${e.message}`);
   786→  }
   787→
   788→  // Strip code fences if present
   789→  articleContent = articleContent.replace(/^```(?:markdown)?\s*/i, "").replace(/\s*```\s*$/i, "");
   790→
   791→  // === Post-processing: clean "h1/h2/h3/H1/H2/H3" literal text from headings ===
   792→  // The AI sometimes literally writes "### h3 سوال" or "### H3: سوال" instead of "### سوال".
   793→  // این regex تمام موارد را تمیز می‌کند.
   794→  const beforeClean = articleContent;
   795→  articleContent = articleContent.replace(
   796→    /^(#{1,6})\s*[hH][1-6][:：\-\s]+/gm,
   797→    "$1 "
   798→  );
   799→  if (articleContent !== beforeClean) {
   800→    log(ctx, "info", `🧹 پاک‌سازی h1/h2/h3 literal از heading‌ها انجام شد.`);
   801→  }
   802→
   803→  // === Post-processing: fix broken internal link patterns ===
   804→  // AI گاهی از فرمت اشتباه /article/slug یا /article// یا /blog استفاده می‌کند.
   805→  // این Step همه موارد را به فرمت صحیح تبدیل می‌کند.
   806→  const beforeLinkFix = articleContent;
   807→  // 1) /article// (double slash) → / (e.g. /article//?tool=tdee → /?tool=tdee)
   808→  articleContent = articleContent.replace(/\/article\/\//g, "/");
   809→  // 2) Markdown links ](/article/slug) → ](/?article=slug)
   810→  articleContent = articleContent.replace(/\]\(\/article\/([^)]+)\)/g, (match, rest: string) => {
   811→    if (rest.includes("?")) {
   812→      const [slug, query] = rest.split("?");
   813→      return `](/?article=${slug}&${query})`;
   814→    }
   815→    return `](/?article=${rest})`;
   816→  });
   817→  // 3) ](/blog) → ](/?screen=articles)
   818→  articleContent = articleContent.replace(/\]\(\/blog\)/g, "](/?screen=articles)");
   819→  // 4) standalone /blog references (not in markdown links)
   820→  articleContent = articleContent.replace(/(^|[\s(])\/blog(?=[\s)]|$)/g, "$1/?screen=articles");
   821→  if (articleContent !== beforeLinkFix) {
   822→    log(ctx, "info", `🔗 اصلاح لینک‌های داخلی شکسته (double-slash, /article/slug, /blog) انجام شد.`);
   823→  }
   824→
   825→  // === Post-processing safety net: insert a paragraph between consecutive images ===
   826→  // حتی اگر AI دستورالعمل «دو تصویر پشت سر هم نده» را نادیده بگیرد، این Step اطمینان می‌دهد
   827→  // که هیچ دو تصویری بدون متن بینشان در خروجی نهایی نباشد.
   828→  const beforeImgFix = articleContent;
   829→  let iterations = 0;
   830→  while (iterations < 10) {
   831→    const next = articleContent.replace(
   832→      /(!\[[^\]]*\]\([^\)]+\))[\s\n]+(!\[[^\]]*\]\([^\)]+\))/g,
   833→      "$1\n\nبرای درک بهتر، تصویر زیر نکات کلیدی را نشان می‌دهد:\n\n$2"
   834→    );
   835→    if (next === articleContent) break;
   836→    articleContent = next;
   837→    iterations++;
   838→  }
   839→  if (articleContent !== beforeImgFix) {
   840→    log(ctx, "info", `🖼 درج پاراگراف انتقالی بین ${iterations} جفت تصویر متوالی انجام شد.`);
   841→  }
   842→
   843→  const wordCount = articleContent.trim().split(/\s+/).length;
   844→  log(ctx, "info", `📄 محتوای مقاله تولید شد (${wordCount} کلمه) — شروع تولید تصاویر`);
   845→
   846→  // ─── Generate cover image ───
   847→  // الگوی یکدست برای تمام کاورها: سبک فیتاپ با تم نارنجی-طلایی
   848→  let coverUrl = "";
   849→  try {
   850→    const coverPrompt =
   851→      plan.coverImagePrompt ||
   852→      `Professional fitness photograph related to "${plan.keyword}", dramatic lighting, dark background with orange-gold accents, athletic aesthetic, photorealistic, high quality, no text, no watermark, cinematic style, warm orange tone, fitness magazine cover style`;
   853→    log(ctx, "info", `🖼 تولید تصویر کاور با الگوی یکدست فیتاپ...`);
   854→    const coverImg = await generateImage({
   855→      prompt: coverPrompt,
   856→      aspectRatio: "16:9" as AspectRatio,
   857→      timeoutMs: 120000,
   858→    });
   859→    const processed = await processAndSaveArticleImage({
   860→      buffer: coverImg.buffer,
   861→      articleSlug: plan.slug,
   862→      descriptiveName: plan.keyword.replace(/\s+/g, "-").slice(0, 40),
   863→    });

   865→    log(
   866→      ctx,
   867→      "success",
   868→      `✅ تصویر کاور تولید و بهینه شد: ${coverUrl} (${Math.round(processed.cover.bytes / 1024)}KB)`
   869→    );
   870→  } catch (e: any) {
   871→    log(ctx, "warn", `تولید تصویر کاور شکست خورد: ${e.message}`);
   872→  }
   873→
   874→  // ─── Generate inline images ───
   875→  // Find IMAGE_PLACEHOLDER_N in content
   876→  const placeholderRegex = /!\[([^\]]*)\]\(IMAGE_PLACEHOLDER_(\d+)\)/g;
   877→  const placeholders: { alt: string; index: number; fullMatch: string }[] = [];
   878→  let match;
   879→  while ((match = placeholderRegex.exec(articleContent)) !== null) {
   880→    placeholders.push({
   881→      alt: match[1],
   882→      index: parseInt(match[2], 10),
   883→      fullMatch: match[0],
   884→    });
   885→  }
   886→
   887→  // Generate inline images (max 3)
   888→  const inlinePrompts = plan.inlineImagePrompts || [];

   890→    const ph = placeholders[i];
   891→    const prompt =
   892→      inlinePrompts[i] ||
   893→      `Photorealistic fitness image illustrating: ${ph.alt || plan.keyword}, dramatic lighting, dark background with orange-gold accents, athletic aesthetic, high quality, cinematic style, warm orange tone, no text, no watermark`;
   894→    try {
   895→      log(ctx, "info", `🖼 تولید تصویر داخل متن ${i + 1}/${placeholders.length}...`);
   896→      const img = await generateImage({
   897→        prompt,
   898→        aspectRatio: "16:9" as AspectRatio,
   899→        timeoutMs: 120000,

   902→        buffer: img.buffer,
   903→        articleSlug: plan.slug,
   904→        descriptiveName: ph.alt.slice(0, 40),
   905→        index: ph.index,
   906→      });

   908→      articleContent = articleContent.replace(
   909→        ph.fullMatch,
   910→        `![${ph.alt}](${processed.url})`
   911→      );
   912→      log(
   913→        ctx,
   914→        "success",
   915→        `✅ تصویر داخل متن ${i + 1} تولید و جایگزین شد: ${processed.url}`

   917→    } catch (e: any) {
   918→      log(ctx, "warn", `تولید تصویر داخل متن ${i + 1} شکست خورد: ${e.message}`);
   919→      // Remove the placeholder on failure
   920→      articleContent = articleContent.replace(ph.fullMatch, "");
   921→    }
   922→  }
   923→
   924→  // Remove any remaining placeholders
   925→  articleContent = articleContent.replace(
   926→    /!\[[^\]]*\]\(IMAGE_PLACEHOLDER_\d+\)/g,
   927→    ""
   928→  );
   929→
   930→  // ─── Generate SEO fields ───
   931→  const seoSystem = `تو متخصص سئو هستی. فیلدهای سئو برای یک مقاله ورزشی/تغذیه فارسی تولید کن. فقط JSON.
   932→مهم: همه فیلدها باید فارسی و حاوی کلمه کلیدی اصلی یا ثانویه باشند.
   933→seoTitle باید ترغیب‌کننده و قابل کلیک باشد (CTR بالا) و حداکثر ۶۰ کاراکتر باشد.
   934→seoDescription باید در سرچ نتایج گوگل جذاب به‌نظر برسد و حداکثر ۱۶۰ کاراکتر باشد.
   935→metaKeywords باید ۵ تا ۸ کلمه کلیدی مرتبط (با کاما جدا شود) باشد.`;
   936→  const seoUser = `برای مقاله زیر فیلدهای سئو بنویس:
   937→عنوان: ${plan.title}
   938→کلمه کلیدی اصلی: ${plan.keyword}
   939→کلمات کلیدی ثانویه: ${plan.secondaryKeywords.join("، ")}
   940→خلاصه: ${plan.excerpt}
   941→
   942→JSON:
   943→{
   944→  "seoTitle": "عنوان سئو حداکثر ۶۰ کاراکتر (شامل کلمه کلیدی)",
   945→  "seoDescription": "توضیحات متا حداکثر ۱۶۰ کاراکتر (شامل کلمه کلیدی، ترغیب‌کننده)",
   946→  "metaKeywords": "کلمه۱, کلمه۲, کلمه۳, کلمه۴, کلمه۵"
   947→}`;
   948→
   949→  let seo = { seoTitle: "", seoDescription: "", metaKeywords: "" };
   950→  try {
   951→    const seoContent = await callLlm(seoSystem, seoUser, 500);
   952→    const parsed = extractJson(seoContent);
   953→    if (parsed) seo = { ...seo, ...parsed };
   954→  } catch {
   955→    // keep defaults
   956→  }
   957→
   958→  // ─── اعمال محدودیت‌های سخت (hard limits) روی فیلدهای سئو ───
   959→  // حتی اگر LLM مقدار طولانی‌تری برگرداند، آن را کوتاه می‌کنیم تا با
   960→  // بهترین شیوه‌های گوگل هم‌خوانی داشته باشد.
   961→  if (!seo.seoTitle) seo.seoTitle = plan.title;
   962→  seo.seoTitle = seo.seoTitle.slice(0, 60);
   963→
   964→  if (!seo.seoDescription) seo.seoDescription = plan.excerpt;
   965→  seo.seoDescription = seo.seoDescription.slice(0, 160);
   966→
   967→  if (!seo.metaKeywords) {
   968→    seo.metaKeywords = [plan.keyword, ...plan.secondaryKeywords.slice(0, 4)].join(", ");
   969→  }
   970→  // محدود کردن metaKeywords به ۸ کلمه کلیدی (و حذف موارد خالی/تکراری)
   971→  const kwList = seo.metaKeywords
   972→    .split(",")
   973→    .map((k) => k.trim())
   974→    .filter((k) => k.length > 0);
   975→  // حذف تکراری‌ها (case-insensitive)
   976→  const seenKw = new Set<string>();
   977→  const dedupKw: string[] = [];
   978→  for (const k of kwList) {
   979→    const lower = k.toLowerCase();
   980→    if (!seenKw.has(lower)) {
   981→      seenKw.add(lower);
   982→      dedupKw.push(k);
   983→    }
   984→  }
   985→  seo.metaKeywords = dedupKw.slice(0, 8).join(", ");
   986→
   987→  // ─── Ensure FAQ section exists in content (for FAQPage JSON-LD) ───
   988→  // اگر مقاله FAQ نداشت، یک بخش FAQ با ۳-۴ سؤال متداول اضافه می‌کنیم
   989→  // تا article-page.tsx بتواند FAQPage JSON-LD تولید کند.
   990→  if (!/##\s+سوالات\s+متداول|##\s+پرسش.*متداول|##\s+FAQ/i.test(articleContent)) {
   991→    const faqSection = `
   992→
   993→## سوالات متداول
   994→
   995→### ${plan.keyword} چیست؟
   996→
   997→${plan.keyword} یکی از مفاهیم کلیدی در مسیر تناسب اندام و سلامتی است. در این مقاله به‌طور کامل توضیح دادیم که چرا برای رسیدن به هدف ورزشی شما اهمیت دارد و چطور می‌توانید آن را به‌درستی پیاده کنید.
   998→
   999→### چطور می‌توانم ${plan.keyword} را در برنامه تمرینی خود بگنجانم؟
  1000→
  1001→با استفاده از اصولی که در این مقاله توضیح دادیم و با کمک مربی هوشمند فیتاپ، می‌توانید ${plan.keyword} را به‌صورت شخصی‌سازی‌شده در برنامه خود داشته باشید. کافیست برنامه اختصاصی فیتاپ را تهیه کنید.
  1002→
  1003→### آیا ${plan.keyword} برای مبتدی‌ها مناسب است؟
  1004→
  1005→بله — با رعایت نکات ایمنی و شروع تدریجی، ${plan.keyword} برای ورزشکاران مبتدی هم مناسب است. توصیه می‌کنیم قبل از شروع، با مربی هوشمند فیتاپ مشورت کنید تا برنامه‌ای متناسب با سطح شما طراحی شود.
  1006→`;
  1007→    articleContent = articleContent.trimEnd() + "\n" + faqSection;
  1008→    log(ctx, "info", `📋 بخش FAQ به مقاله اضافه شد (برای FAQPage JSON-LD).`);
  1009→  }
  1010→
  1011→  // ─── Compute reading time ───
  1012→  const readingMinutes = Math.max(1, Math.ceil(wordCount / 200));
  1013→
  1014→  // ─── Compute scheduled publish date (space out over weeks per strategy calendar) ───
  1015→  // The article is saved as "draft" with a scheduledAt in the future. The
  1016→  // /api/cron/publish-scheduled endpoint will flip it to "published" when the
  1017→  // scheduledAt arrives. This gives a steady drip of fresh content rather than
  1018→  // dumping N articles on the same day.
  1019→  const scheduledAt = await computeScheduledAt(plan, strategy, ctx.publishImmediately === true);
  1020→
  1021→  // ─── SEO fields: canonical URL, OG image, robots ───
  1022→  // canonical URL: همیشه مطلق (https://...) باشد تا گوگل آن را به‌عنوان canonical
  1023→  // صفحه قبول کند. در غیر این صورت، مقاله‌های تولیدشده canonical خالی دارند و
  1024→  // article-page.tsx به‌صورت fallback از ?article=slug استفاده می‌کند — اما
  1025→  // ذخیره URL مطلق در DB تمرین بهتری است و با مقاله‌های دستی هم‌خوانی دارد.
  1026→  const siteUrl = (process.env.NEXT_PUBLIC_SITE_URL || "https://fittup.ir").replace(/\/$/, "");
  1027→  const canonicalUrl = `${siteUrl}/?article=${plan.slug}`;
  1028→  const ogImage = coverUrl; // OG image همان تصویر کاور است
  1029→  const robots = "index,follow";
  1030→
  1031→  // ─── Save article to DB ───
  1032→  const article = await db.article.create({
  1033→    data: {
  1034→      title: plan.title,
  1035→      slug: plan.slug,
  1036→      excerpt: plan.excerpt,
  1037→      content: articleContent,
  1038→      category: plan.category,
  1039→      tags: [plan.keyword, ...plan.secondaryKeywords].slice(0, 8).join(", "),
  1040→      // If a schedule was computed → keep draft + scheduledAt; otherwise publish now.
  1041→      status: scheduledAt ? "draft" : "published",
  1042→      scheduledAt: scheduledAt ?? undefined,
  1043→      authorId: ctx.adminId,
  1044→      coverImage: coverUrl,

  1046→      seoDescription: seo.seoDescription,
  1047→      metaKeywords: seo.metaKeywords,
  1048→      ogImage,
  1049→      canonicalUrl,
  1050→      robots,
  1051→      readingMinutes,
  1052→    },
  1053→  });
  1054→
  1055→  // Update plan as published
  1056→  if (plan.id) {
  1057→    await db.seoArticlePlan.update({
  1058→      where: { id: plan.id },
  1059→      data: { status: "published", articleId: article.id },
  1060→    });
  1061→  }
  1062→
  1063→  if (scheduledAt) {
  1064→    log(
  1065→      ctx,
  1066→      "success",
  1067→      `🗓 مقاله زمان‌بندی شد: "${plan.title}" — ${wordCount} کلمه، ${readingMinutes} دقیقه مطالعه، ${coverUrl ? "با تصویر" : "بدون تصویر"} — انتشار: ${scheduledAt.toLocaleDateString("fa-IR")} ساعت ${scheduledAt.toLocaleTimeString("fa-IR", { hour: "2-digit", minute: "2-digit" })}`
  1068→    );
  1069→  } else {
  1070→    log(
  1071→      ctx,
  1072→      "success",
  1073→      `✅ مقاله منتشر شد: "${plan.title}" — ${wordCount} کلمه، ${readingMinutes} دقیقه مطالعه، ${coverUrl ? "با تصویر" : "بدون تصویر"}`
  1074→    );
  1075→  }
  1076→
  1077→  return { articleId: article.id, coverUrl, wordCount, scheduledAt };
  1078→}

  1080→// ─── Main entry point: run the agent ───
  1081→export async function runSeoAgent(opts: {
  1082→  runId: string;
  1083→  adminId: string;
  1084→  mode: "full" | "continue" | "strategy_only";
  1085→  count: number;
  1086→  /**
  1087→   * If true, generated articles are published immediately instead of being
  1088→   * scheduled for future publishing. Defaults to false (schedule articles).
  1089→   */
  1090→  publishImmediately?: boolean;
  1091→}): Promise<RunContext> {
  1092→  const ctx: RunContext = {
  1093→    runId: opts.runId,
  1094→    mode: opts.mode,
  1095→    count: opts.count,
  1096→    logs: [],
  1097→    adminId: opts.adminId,
  1098→    startedAt: Date.now(),
  1099→    successCount: 0,
  1100→    failCount: 0,
  1101→    articles: [],
  1102→    errors: [],
  1103→    publishImmediately: opts.publishImmediately === true,
  1104→  };
  1105→
  1106→  const startedRun = await db.seoAgentRun.update({
  1107→    where: { id: opts.runId },
  1108→    data: {
  1109→      status: "running",
  1110→      mode: opts.mode,
  1111→      requestedCount: opts.count,
  1112→      startedAt: new Date(),
  1113→    },
  1114→  }).catch(() => null);
  1115→
  1116→  try {
  1117→    // Step 1: Always analyze site first (for accurate internal linking)
  1118→    const site = await analyzeSite(ctx);
  1119→
  1120→    // Step 2: Generate/refresh strategy (in full mode) or load existing
  1121→    let strategy: SeoStrategyContent;
  1122→    if (opts.mode === "full") {
  1123→      strategy = await generateStrategy(ctx, site);
  1124→    } else {
  1125→      const existing = await db.seoStrategy.findFirst({
  1126→        where: { isActive: true },
  1127→        orderBy: { version: "desc" },
  1128→      });
  1129→      if (existing) {
  1130→        strategy = JSON.parse(existing.content) as SeoStrategyContent;
  1131→        log(ctx, "info", `📋 استفاده از استراتژی موجود (نسخه ${existing.version})`);
  1132→      } else {
  1133→        // No strategy yet — must generate
  1134→        log(ctx, "warn", "استراتژی موجود نیست — تولید استراتژی جدید");
  1135→        strategy = await generateStrategy(ctx, site);
  1136→      }
  1137→    }
  1138→
  1139→    if (opts.mode === "strategy_only") {
  1140→      log(ctx, "success", "🏁 حالت strategy_only — استراتژی تولید شد، بدون تولید مقاله");
  1141→      await finishRun(ctx, "completed");
  1142→      return ctx;
  1143→    }
  1144→
  1145→    // Step 3: Plan articles (or use existing planned ones)
  1146→    let plans: ArticlePlan[] = [];
  1147→
  1148→    // First: use existing planned articles (continue mode)
  1149→    const existingPlans = await db.seoArticlePlan.findMany({
  1150→      where: { status: "planned" },
  1151→      orderBy: { priority: "desc" },
  1152→      take: opts.count,
  1153→    });
  1154→
  1155→    if (existingPlans.length >= opts.count && opts.mode === "continue") {
  1156→      plans = existingPlans.map((p) => ({
  1157→        id: p.id,
  1158→        keyword: p.keyword,
  1159→        title: p.title,
  1160→        slug: p.slug,
  1161→        category: p.category,
  1162→        excerpt: p.excerpt,
  1163→        pillar: p.pillar,
  1164→        secondaryKeywords: p.secondaryKeywords.split(", ").filter(Boolean),
  1165→        internalLinks: JSON.parse(p.internalLinks || "[]"),
  1166→        coverImagePrompt: p.coverImagePrompt,
  1167→        inlineImagePrompts: [],
  1168→      }));
  1169→      log(ctx, "info", `📋 استفاده از ${plans.length} مقاله برنامه‌ریزی‌شده موجود در صف`);
  1170→    } else {
  1171→      // Plan new articles
  1172→      const needed = opts.count - existingPlans.length;
  1173→      if (needed > 0) {
  1174→        log(ctx, "info", `📅 نیاز به ${needed} مقاله جدید — برنامه‌ریزی...`);
  1175→        await planArticles(ctx, strategy, site, Math.max(needed, opts.count));
  1176→      }
  1177→      // Now fetch all planned
  1178→      const allPlanned = await db.seoArticlePlan.findMany({
  1179→        where: { status: "planned" },
  1180→        orderBy: { priority: "desc" },
  1181→        take: opts.count,
  1182→      });
  1183→      plans = allPlanned.map((p) => ({
  1184→        id: p.id,
  1185→        keyword: p.keyword,
  1186→        title: p.title,
  1187→        slug: p.slug,
  1188→        category: p.category,
  1189→        excerpt: p.excerpt,
  1190→        pillar: p.pillar,
  1191→        secondaryKeywords: p.secondaryKeywords.split(", ").filter(Boolean),
  1192→        internalLinks: JSON.parse(p.internalLinks || "[]"),
  1193→        coverImagePrompt: p.coverImagePrompt,
  1194→        inlineImagePrompts: [],
  1195→      }));
  1196→    }
  1197→
  1198→    if (plans.length === 0) {
  1199→      log(ctx, "warn", "هیچ مقاله‌ای برای تولید یافت نشد");
  1200→      await finishRun(ctx, "completed");
  1201→      return ctx;
  1202→    }
  1203→
  1204→    log(ctx, "info", `🚀 شروع تولید ${plans.length} مقاله`);
  1205→
  1206→    // Step 4: Generate each article (sequentially to avoid rate limits)
  1207→    // Re-analyze site before each article to ensure accurate internal linking (new articles get added)
  1208→    for (let i = 0; i < plans.length; i++) {
  1209→      const plan = plans[i];
  1210→      log(ctx, "info", `━━━ مقاله ${i + 1}/${plans.length} ━━━`);
  1211→      try {
  1212→        // Refresh site analysis (so newly-published articles can be linked)
  1213→        const freshSite = i === 0 ? site : await analyzeSite(ctx);
  1214→        const result = await generateArticle(ctx, plan, freshSite, strategy);
  1215→        ctx.articles.push({
  1216→          id: result.articleId,
  1217→          title: plan.title,
  1218→          slug: plan.slug,
  1219→          keyword: plan.keyword,
  1220→          coverImage: result.coverUrl ? "✓" : "✗",
  1221→          words: result.wordCount,
  1222→          scheduledAt: result.scheduledAt ? result.scheduledAt.toISOString() : null,
  1223→        });
  1224→        ctx.successCount++;
  1225→      } catch (e: any) {
  1226→        log(ctx, "error", `مقاله "${plan.title}" شکست خورد: ${e.message}`);
  1227→        ctx.errors.push(`${plan.title}: ${e.message}`);
  1228→        ctx.failCount++;
  1229→        if (plan.id) {
  1230→          await db.seoArticlePlan.update({
  1231→            where: { id: plan.id },
  1232→            data: { status: "failed", errorMessage: e.message.slice(0, 500) },
  1233→          }).catch(() => null);
  1234→        }
  1235→      }
  1236→      // Persist progress to DB after each article
  1237→      await persistRunProgress(ctx);
  1238→    }
  1239→
  1240→    log(
  1241→      ctx,
  1242→      "success",
  1243→      `🏁 پایان اجرا — ${ctx.successCount} موفق، ${ctx.failCount} ناموفق`
  1244→    );
  1245→
  1246→    await finishRun(ctx, ctx.failCount === 0 ? "completed" : "partial");
  1247→    return ctx;
  1248→  } catch (e: any) {
  1249→    log(ctx, "error", `خطای بحرانی در اجرای ایجنت: ${e.message}`);
  1250→    ctx.errors.push(`CRITICAL: ${e.message}`);
  1251→    await finishRun(ctx, "failed");
  1252→    return ctx;
  1253→  }
  1254→}

  1316→export async function startBackgroundRun(opts: {
  1317→  adminId: string;
  1318→  mode: "full" | "continue" | "strategy_only";
  1319→  count: number;
  1320→  publishImmediately?: boolean;
  1321→}): Promise<{ runId: string; alreadyRunning: boolean }> {
  1322→  if (isAgentRunning()) {
  1323→    return { runId: getCurrentRunId() || "", alreadyRunning: true };
  1324→  }
  1325→  // Create run record
  1326→  const run = await db.seoAgentRun.create({
  1327→    data: {
  1328→      mode: opts.mode,
  1329→      requestedCount: opts.count,
  1330→      status: "running",
  1331→      startedAt: new Date(),
  1332→    },
  1333→  });
  1334→  globalThis.__seoAgentRunning = true;
  1335→  globalThis.__seoAgentRunId = run.id;
  1336→
  1337→  // Fire and forget — run in background
  1338→  runSeoAgent({
  1339→    runId: run.id,
  1340→    adminId: opts.adminId,
  1341→    mode: opts.mode,
  1342→    count: opts.count,
  1343→    publishImmediately: opts.publishImmediately,
  1344→  })
  1345→    .catch((e) => {
  1346→      console.error("[SEO-AGENT] background run crashed:", e);
  1347→    })
  1348→    .finally(() => {
  1349→      globalThis.__seoAgentRunning = false;
  1350→      globalThis.__seoAgentRunId = undefined;
  1351→    });
  1352→
  1353→  return { runId: run.id, alreadyRunning: false };
  1354→}
