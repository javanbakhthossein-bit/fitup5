   291→export function AdminOverlay({ standalone = false }: { standalone?: boolean } = {}) {
   292→  const { setOverlay, reset } = useAppStore();
   293→  const [tab, setTab] = useState<AdminTab>("dashboard");
   294→  const [showSettings, setShowSettings] = useState(false);
   295→  const [permissions, setPermissions] = useState<AdminPermissions>(ALL_TRUE_PERMISSIONS);
   296→  const [permsLoaded, setPermsLoaded] = useState(false);
   297→
   298→  // Fetch the current admin's permissions on mount — used to filter visible tabs
   299→  useEffect(() => {
   300→    (async () => {
   301→      try {
   302→        const res = await fetch("/api/admin/permissions", { cache: "no-store" });
   303→        if (res.ok) {
   304→          const data = await res.json();
   305→          if (data?.permissions) setPermissions(data.permissions as AdminPermissions);
   306→        }
   307→      } catch {
   308→        // fall back to all-true if the call fails (best-effort)
   309→      } finally {
   310→        setPermsLoaded(true);
   311→      }
   312→    })();
   313→  }, []);
   314→
   315→  async function handleLogout() {
   316→    try {
   317→      await fetch("/api/auth/logout", { method: "POST" });
   318→      reset();
   319→      toast.success("از پنل مدیریت خارج شدید");
   320→    } catch {
   321→      toast.error("خطا در خروج");
   322→    }
   323→  }
   324→
   325→  // All possible tabs — we filter the visible ones based on permissions
   326→  const allTabs: { id: AdminTab; label: string; icon: any; perm: PermissionKey }[] = [
   327→    { id: "dashboard", label: "داشبورد", icon: LayoutDashboard, perm: "canViewDashboard" },
   328→    { id: "users", label: "کاربران", icon: Users, perm: "canManageUsers" },
   329→    { id: "finance", label: "مالی و تراکنش‌ها", icon: Wallet, perm: "canViewFinance" },
   330→    { id: "programs", label: "صف برنامه‌ها", icon: ListChecks, perm: "canManagePrograms" },
   331→    { id: "checkups", label: "چکاپ‌ها", icon: ClipboardCheck, perm: "canManageCheckups" },
   332→    { id: "articles", label: "مقالات", icon: Newspaper, perm: "canManageArticles" },
   333→    { id: "head_codes", label: "کدهای تحلیلی", icon: Code2, perm: "canManageHeadCodes" },
   334→    { id: "terms", label: "قوانین", icon: ShieldCheck, perm: "canManageTerms" },
   335→    { id: "tickets", label: "تیکت‌ها", icon: MessageSquare, perm: "canManageTickets" },
   336→    { id: "copilot", label: "دستیار هوشمند", icon: Sparkles, perm: "canUseCopilot" },
   337→    { id: "admins", label: "مدیریت ادمین‌ها", icon: ShieldCheck, perm: "canManageAdmins" },
   338→    { id: "domain", label: "دامنه و رکوردها", icon: Globe, perm: "canManageHeadCodes" },
   339→    { id: "seo", label: "سئو هوشمند", icon: Rocket, perm: "canManageArticles" },
   340→  ];
   341→
   342→  // Filter tabs by permission; always show at least the dashboard
   343→  const tabs = allTabs.filter((t) => permissions[t.perm]);
   344→
   345→  // If the current tab is not visible (e.g. perms changed), fall back to first available
   346→  useEffect(() => {
   347→    if (permsLoaded && tabs.length > 0 && !tabs.some((t) => t.id === tab)) {
   348→      setTab(tabs[0].id);
   349→    }
   350→  }, [permsLoaded, permissions, tabs, tab]);
   351→
   352→  return (
   353→    <div className={`flex flex-col h-full bg-background ${standalone ? "fixed inset-0 z-50" : ""}`}>
   354→      {/* Header */}
   355→      <div className="flex items-center justify-between p-4 border-b glass-dark">
   356→        <div className="flex items-center gap-2">
   357→          <Shield className="w-5 h-5 text-primary" />
   358→          <h2 className="font-bold">پنل مدیریت فیتاپ</h2>
   359→        </div>
   360→        <div className="flex items-center gap-1">
   361→          <Button variant="ghost" size="sm" onClick={() => setShowSettings(true)} className="rounded-xl gap-1.5 text-xs">
   362→            <SettingsIcon className="w-4 h-4" />
   363→            <span className="hidden sm:inline">تنظیمات سایت</span>
   364→          </Button>

   366→            <Button variant="ghost" size="sm" onClick={handleLogout} className="rounded-xl gap-1.5 text-xs text-red-600 hover:text-red-700 hover:bg-red-50">
   367→              <LogOut className="w-4 h-4" />
   368→              <span className="hidden sm:inline">خروج</span>
   369→            </Button>
   370→          ) : (
   371→            <Button variant="ghost" size="icon" onClick={() => setOverlay(null)} className="rounded-full">
   372→              <X className="w-5 h-5" />
   373→            </Button>
   374→          )}
   375→        </div>
   376→      </div>
   377→
   378→      {/* Tabs */}
   379→      <div className="flex gap-1 p-2 border-b overflow-x-auto no-scrollbar glass-dark">

   402→            {tab === "checkups" && <CheckupsTab />}
   403→            {tab === "articles" && <ArticlesTab />}
   404→            {tab === "head_codes" && <HeadCodesTab />}
   405→            {tab === "terms" && <TermsTab />}
   406→            {tab === "tickets" && <TicketsTab />}
   407→            {tab === "copilot" && <CopilotTab />}

   409→            {tab === "domain" && <DomainTab />}
   410→            {tab === "seo" && <SeoAgentTab />}

   415→      {/* Site Settings Dialog */}

   773→function ChargeWalletDialog({ user, onClose, onDone }: { user: any; onClose: () => void; onDone: () => void }) {
   774→  const [amount, setAmount] = useState("");
   775→  const [desc, setDesc] = useState("");
   776→  const [saving, setSaving] = useState(false);
   777→
   778→  async function charge() {
   779→    if (!amount || Number(amount) <= 0) return;
   780→    setSaving(true);
   781→    try {
   782→      const res = await fetch("/api/admin/wallet-charge", {
   783→        method: "POST", headers: { "Content-Type": "application/json" },
   784→        body: JSON.stringify({ userId: user.id, amount: Number(amount), description: desc }),
   785→      });
   786→      if (!res.ok) throw new Error();
   787→      toast.success(`کیف پول ${toPersianDigits(formatToman(Number(amount)))} ت شارژ شد`);
   788→      onDone();
   789→    } catch { toast.error("خطا در شارژ"); } finally { setSaving(false); }
   790→  }
   791→
   792→  return (
   793→    <Dialog open onOpenChange={onClose}>
   794→      <DialogContent dir="rtl" className="max-w-sm">
   795→        <DialogHeader><DialogTitle>شارژ کیف پول — {user.name || user.mobile}</DialogTitle></DialogHeader>
   796→        <div className="space-y-3">
   797→          <div className="p-2 rounded-lg bg-muted/40 text-xs text-center">موجودی فعلی: <b className="font-stat">{toPersianDigits(formatToman(user.walletBalance || 0))} ت</b></div>
   798→          <div>
   799→            <Label className="mb-1 block">مبلغ (تومان)</Label>

   803→            <Label className="mb-1 block">توضیحات (اختیاری)</Label>
   804→            <Input value={desc} onChange={(e) => setDesc(e.target.value)} placeholder="شارژ تستی / پشتیبانی" className="rounded-xl" />
   805→          </div>
   806→          <div className="flex gap-2 flex-wrap">
   807→            {[100000, 500000, 1000000].map(v => (
   808→              <button key={v} onClick={() => setAmount(String(v))} className="text-xs px-3 py-1.5 rounded-lg bg-muted hover:bg-primary hover:text-primary-foreground transition font-stat">
   809→                {toPersianDigits(formatToman(v))}
   810→              </button>
   811→            ))}
