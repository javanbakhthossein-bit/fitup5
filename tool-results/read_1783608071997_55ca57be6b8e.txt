    20→export function getAvalaiClient(): OpenAI {
    21→  if (!_avalaiClient) {
    22→    _avalaiClient = new OpenAI({
    23→      apiKey: process.env.AVALAI_API_KEY || "placeholder-for-build",
    24→      baseURL: process.env.AVALAI_BASE_URL || "https://api.avalai.ir/v1",
    25→    });
    26→  }
    27→  return _avalaiClient;
    28→}

  1637→// Parse JSON from LLM content (handles markdown code fences + HTML error pages gracefully)
  1638→function parseJsonFromContent(content: string): any {
  1639→  if (!content || typeof content !== "string") {
  1640→    return { days: [], meals: [], notes: "پاسخ هوش مصنوعی خالی بود." };
  1641→  }
  1642→
  1643→  let cleaned = content.trim();
  1644→
  1645→  // Detect HTML error pages (gateway error, 502/504, AvalAI returning HTML)
  1646→  // If the content looks like HTML and contains no JSON braces, bail out gracefully.
  1647→  const looksLikeHtml = /^\s*<(?:html|!doctype|head|body|h1|div|p)\b/i.test(cleaned) ||
  1648→    (cleaned.startsWith("<") && cleaned.includes("</") && !cleaned.includes("{"));
  1649→  if (looksLikeHtml) {
  1650→    console.error("[parseJsonFromContent] AI returned HTML (likely an error page):", cleaned.slice(0, 200));
  1651→    return { days: [], meals: [], notes: "خطا در پاسخ هوش مصنوعی. لطفاً دوباره تلاش کنید." };
  1652→  }
  1653→
  1654→  // Extract JSON from ```json ... ``` or ``` ... ```
  1655→  const fenceMatch = cleaned.match(/```(?:json)?\s*([\s\S]*?)```/);
  1656→  if (fenceMatch) {
  1657→    cleaned = fenceMatch[1].trim();
  1658→  }
  1659→
  1660→  // Find first { and last }
  1661→  const first = cleaned.indexOf("{");
  1662→  const last = cleaned.lastIndexOf("}");
  1663→  if (first !== -1 && last !== -1 && last > first) {
  1664→    cleaned = cleaned.slice(first, last + 1);
  1665→  }
  1666→
  1667→  try {
  1668→    return JSON.parse(cleaned);
  1669→  } catch (err) {
  1670→    console.error("[parseJsonFromContent] JSON parse failed. Raw content:", content.slice(0, 300), err);
  1671→    return { days: [], meals: [], notes: "خطا در پردازش پاسخ هوش مصنوعی." };
  1672→  }
