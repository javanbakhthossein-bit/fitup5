   162→// Build user context string from onboarding + plan tier
   163→export function buildUserContext(data: OnboardingData, planName?: Plan | null): string {
   164→  const bmr =
   165→    data.gender === "male"
   166→      ? 10 * data.weight + 6.25 * data.height - 5 * data.age + 5
   167→      : 10 * data.weight + 6.25 * data.height - 5 * data.age - 161;
   168→  const caps = getCapabilities(planName ?? null);
   169→  const planLabel = planName ? PLAN_LABELS[planName] : "بدون پلن (Basic)";
   170→  return `اطلاعات کاربر:
   171→- جنسیت: ${GENDER_LABELS[data.gender]}
   172→- سن: ${data.age} سال
   173→- قد: ${data.height} سانتی‌متر
   174→- وزن: ${data.weight} کیلوگرم
   175→- وزن هدف: ${data.targetWeight ?? "نامشخص"} کیلوگرم
   176→- هدف اصلی: ${GOAL_LABELS[data.goal]}
   177→- سطح فعالیت: ${ACTIVITY_LABELS[data.activityLevel]}
   178→- روزهای تمرین در هفته: ${data.workoutDays} روز${data.workoutDaysList && data.workoutDaysList.length > 0 ? ` (${data.workoutDaysList.join("، ")})` : ""}
   179→- مکان تمرین: ${WORKOUT_PLACE_LABELS[data.workoutPlace]}
   180→- تجهیزات: ${data.equipment.length ? data.equipment.join("، ") : "بدون تجهیزات خاص"}
   181→- سوابق بیماری: ${data.diseases || "ندارد"}
   182→- آسیب‌دیدگی مفصلی/عضلانی: ${data.injuries || "ندارد"}
   183→- حساسیت غذایی: ${data.allergies || "ندارد"}
   184→- نوع رژیم غذایی: ${DIET_LABELS[data.dietType]}
   185→${data.trainingExperience ? `- سابقه ورزشی: ${TRAINING_EXPERIENCE_LABELS[data.trainingExperience]}` : ""}
   186→${data.previousTrainingType ? `- نوع تمرین قبلی: ${data.previousTrainingType}` : ""}
   187→${data.drugAllergies ? `- آلرژی‌های دارویی: ${data.drugAllergies}` : ""}
   188→${data.currentMedications ? `- داروهای مصرفی: ${data.currentMedications}` : ""}
   189→${data.maxLifts ? `- حداکثر وزنه‌ها: ${data.maxLifts}` : ""}
   190→- BMR تخمینی: ${Math.round(bmr)} کالری
   191→- پلن اشتراک کاربر: ${planLabel}
   192→- قابلیت‌های فعال: ${[
   193→    caps.workoutAndNutritionPlan ? "برنامه تمرین+تغذیه" : null,
   194→    caps.supplementsPlan ? "برنامه مکمل" : null,
   195→    caps.periodicCheckups ? "چکاپ دوره‌ای" : null,
   196→    caps.aiChatQuestions !== 0 ? "چت هوشمند" : null,
   197→    caps.mealPhotoAnalysis ? "آنالیز عکس غذا" : null,
   198→    caps.bodyPhotoAnalysis ? "آنالیز عکس بدن" : null,
   199→    caps.videoBodyAnalysis ? "آنالیز ویدیویی بدن" : null,
   200→    caps.techniqueCorrection ? "اصلاح تکنیک" : null,
   201→    caps.bloodTestAnalysis ? "تحلیل آزمایش خون" : null,
   202→  ].filter(Boolean).join("، ")}`;
   203→}

  1027→// Parse JSON from LLM content (handles markdown code fences + HTML error pages gracefully)
  1028→function parseJsonFromContent(content: string): any {
  1029→  if (!content || typeof content !== "string") {
  1030→    return { days: [], meals: [], notes: "پاسخ هوش مصنوعی خالی بود." };
  1031→  }
  1032→
  1033→  let cleaned = content.trim();
  1034→
  1035→  // Detect HTML error pages (gateway error, 502/504, AvalAI returning HTML)
  1036→  // If the content looks like HTML and contains no JSON braces, bail out gracefully.
  1037→  const looksLikeHtml = /^\s*<(?:html|!doctype|head|body|h1|div|p)\b/i.test(cleaned) ||
  1038→    (cleaned.startsWith("<") && cleaned.includes("</") && !cleaned.includes("{"));
  1039→  if (looksLikeHtml) {
  1040→    console.error("[parseJsonFromContent] AI returned HTML (likely an error page):", cleaned.slice(0, 200));
  1041→    return { days: [], meals: [], notes: "خطا در پاسخ هوش مصنوعی. لطفاً دوباره تلاش کنید." };
  1042→  }
  1043→
  1044→  // Extract JSON from ```json ... ``` or ``` ... ```
  1045→  const fenceMatch = cleaned.match(/```(?:json)?\s*([\s\S]*?)```/);
  1046→  if (fenceMatch) {
  1047→    cleaned = fenceMatch[1].trim();
  1048→  }
  1049→
  1050→  // Find first { and last }
  1051→  const first = cleaned.indexOf("{");
  1052→  const last = cleaned.lastIndexOf("}");
  1053→  if (first !== -1 && last !== -1 && last > first) {
  1054→    cleaned = cleaned.slice(first, last + 1);
  1055→  }
  1056→
  1057→  try {
  1058→    return JSON.parse(cleaned);
  1059→  } catch (err) {
  1060→    console.error("[parseJsonFromContent] JSON parse failed. Raw content:", content.slice(0, 300), err);
  1061→    return { days: [], meals: [], notes: "خطا در پردازش پاسخ هوش مصنوعی." };
  1062→  }
  1063→}
