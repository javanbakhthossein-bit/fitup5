   102→   102→  level: RunProgress["level"],
   103→   103→  msg: string
   104→   104→): void {
   105→   105→  const entry: RunProgress = { level, msg, ts: Date.now() };
   106→   106→  ctx.logs.push(entry);

   108→   108→  const prefix =
   109→   109→    level === "error"
   110→   110→      ? "[SEO-AGENT:ERROR]"
   111→   111→      : level === "warn"
   112→   112→      ? "[SEO-AGENT:WARN]"
   113→   113→      : level === "success"
   114→   114→      ? "[SEO-AGENT:OK]"
   115→   115→      : "[SEO-AGENT]";
   116→   116→  console.log(`${prefix} ${msg}`);

   118→   118→  const now = Date.now();
   119→   119→  const lastPersist = (ctx as any)._lastPersist || 0;
   120→   120→  if (ctx.logs.length % 5 === 0 || now - lastPersist > 3000) {
   121→   121→    (ctx as any)._lastPersist = now;
   122→   122→    // fire-and-forget — don't block the agent
   123→   123→    db.seoAgentRun
   124→   124→      .update({
   125→   125→        where: { id: ctx.runId },
   126→   126→        data: {

   129→   129→          failCount: ctx.failCount,
   130→   130→        },
   131→   131→      })
   132→   132→      .catch(() => null);

   136→   136→// ─── JSON extraction helper ───
   137→   137→function extractJson(content: string): any | null {
   138→   138→  if (!content) return null;
   139→   139→  // Try code block first
   140→   140→  const codeBlock = content.match(/```(?:json)?\s*([\s\S]*?)```/);
   141→   141→  const candidate = codeBlock ? codeBlock[1].trim() : content.trim();
   142→   142→  // Find the first balanced { ... }
   143→   143→  const start = candidate.indexOf("{");
   144→   144→  if (start === -1) return null;
   145→   145→  let depth = 0;
   146→   146→  for (let i = start; i < candidate.length; i++) {
   147→   147→    if (candidate[i] === "{") depth++;
   148→   148→    else if (candidate[i] === "}") {
   149→   149→      depth--;
   150→   150→      if (depth === 0) {
   151→   151→        const jsonStr = candidate.slice(start, i + 1);
   152→   152→        try {
   153→   153→          return JSON.parse(jsonStr);
   154→   154→        } catch {
   155→   155→          // try to fix common issues
   156→   156→          try {
   157→   157→            return JSON.parse(
   158→   158→              jsonStr
   159→   159→                .replace(/,\s*}/g, "}")
   160→   160→                .replace(/,\s*]/g, "]")
   161→   161→                .replace(/[\u201C\u201D]/g, '"')
   162→   162→                .replace(/[\u2018\u2019]/g, "'")
   163→   163→            );
   164→   164→          } catch {
   165→   165→            return null;
   166→   166→          }
   167→   167→        }
   168→   168→      }
   169→   169→    }
   170→   170→  }
   171→   171→  return null;
   172→   172→}

   174→   174→async function callLlm(system: string, user: string, maxTokens = 4000): Promise<string> {
   175→   175→  // timeout ۳ دقیقه — برای production که AI کندتر پاسخ می‌دهد
   176→   176→  const timeoutMs = 180000;
   177→   177→  const controller = new AbortController();
   178→   178→  const timeout = setTimeout(() => controller.abort(), timeoutMs);
   179→   179→  try {
   180→   180→    const completion = await getAvalaiClient().chat.completions.create({
   181→   181→      model: TEXT_MODEL,
   182→   182→      messages: [
   183→   183→        { role: "system", content: system },
   184→   184→        { role: "user", content: user },
   185→   185→      ],
   186→   186→      max_tokens: maxTokens,
   187→   187→      temperature: 0.7,
   188→   188→      // @ts-expect-error — OpenAI SDK supports signal
   189→   189→      signal: controller.signal,
   190→   190→    } as any);
   191→   191→    return completion.choices[0]?.message?.content || "";
   192→   192→  } catch (e: any) {
   193→   193→    if (e.name === "AbortError") {
   194→   194→      throw new Error(`AI timeout after ${timeoutMs / 1000}s`);
   195→   195→    }
   196→   196→    throw e;
   197→   197→  } finally {
   198→   198→    clearTimeout(timeout);
   199→   199→  }
   200→   200→}

   203→   203→export async function analyzeSite(ctx: RunContext): Promise<SiteAnalysis> {
   204→   204→  log(ctx, "info", "🔍 شروع تحلیل سایت — استخراج صفحات و مقالات موجود");
   205→   205→  // Static landing pages & tools (همیشه در سایت فیتاپ وجود دارند)
   206→   206→  // مهم: همه مسیرهای معتبر لینک داخلی در سایت فیتاپ. از این لیست برای لینک‌سازی استفاده کن.
   207→   207→  const pages = [
   208→   208→    { url: "/", title: "فیتاپ — مربی هوشمند بدنسازی و تغذیه", type: "landing" as const },
   209→   209→    { url: "/?screen=articles", title: "مقالات ورزشی و تغذیه فیتاپ", type: "list" as const },
   210→   210→    { url: "/?tool=tdee", title: "محاسبه‌گر کالری و TDEE", type: "tool" as const },
   211→   211→    { url: "/?tool=exercises", title: "بانک حرکات ورزشی", type: "tool" as const },
   212→   212→    { url: "/?tool=foods", title: "جدول کالری غذاها", type: "tool" as const },
   213→   213→    { url: "/?screen=auth", title: "ورود / ثبت‌نام فیتاپ", type: "landing" as const },
   214→   214→  ];
   215→   215→
   216→   216→  const articles = await db.article.findMany({
   217→   217→    where: { status: "published" },
   218→   218→    select: {
   219→   219→      id: true,
   220→   220→      title: true,
   221→   221→      slug: true,
   222→   222→      category: true,
   223→   223→      tags: true,
   224→   224→      excerpt: true,
   225→   225→    },
   226→   226→    orderBy: { createdAt: "desc" },
   227→   227→  });
   228→   228→
   229→   229→  // Collect existing keywords from tags
   230→   230→  const existingKeywords = new Set<string>();
   231→   231→  for (const a of articles) {
   232→   232→    if (a.tags) {
   233→   233→      for (const t of a.tags.split(",").map((s) => s.trim()).filter(Boolean)) {
   234→   234→        existingKeywords.add(t);
   235→   235→      }
   236→   236→    }
   237→   237→  }
   238→   238→
   239→   239→  const totalViewsAgg = await db.article.aggregate({
   240→   240→    where: { status: "published" },
   241→   241→    _sum: { views: true },
   242→   242→  });
   243→   243→
   244→   244→  const result: SiteAnalysis = {
   245→   245→    pages,
   246→   246→    articles: articles.map((a) => ({
   247→   247→      id: a.id,
   248→   248→      title: a.title,
   249→   249→      slug: a.slug,
   250→   250→      category: a.category,
   251→   251→      tags: a.tags,
   252→   252→      excerpt: a.excerpt,
   253→   253→    })),
   254→   254→    existingKeywords: Array.from(existingKeywords),
   255→   255→    totalArticles: articles.length,
   256→   256→    totalViews: totalViewsAgg._sum.views || 0,
   257→   257→  };
   258→   258→
   259→   259→  log(
   260→   260→    ctx,
   261→   261→    "success",
   262→   262→    `✅ تحلیل سایت کامل شد: ${result.pages.length} صفحه کلیدی، ${result.totalArticles} مقاله موجود، ${result.existingKeywords.length} کلمه کلیدی موجود، ${result.totalViews} بازدید کل`
   263→   263→  );
   264→   264→  return result;
   265→   265→}

   625→   625→/**
   626→   626→ * Compute the scheduled publish date for a new article.
   627→   627→ *
   628→   628→ * Scheduling rules:
   629→   629→ *  1. If the strategy's contentCalendar has an entry whose keyword matches
   630→   630→ *     (case-insensitive substring) the article's plan keyword, the article is
   631→   631→ *     scheduled for `now + (week * 7 days)`. Week 1 = next week, week 2 = 2
   632→   632→ *     weeks from now, etc. — this spreads articles across the calendar as the
   633→   633→ *     strategy intends.
   634→   634→ *  2. Otherwise, the article is scheduled 7 days after the latest already-scheduled
   635→   635→ *     draft article (so articles never collide on the same day). If no draft is
   636→   636→ *     currently scheduled, it's scheduled 1 day from now as a sensible default.
   637→   637→ *  3. The returned date is always strictly in the future (>= now + 1 hour) so the
   638→   638→ *     cron publisher won't immediately fire on the same run.
   639→   639→ *
   640→   640→ * Returns null when `publishImmediately` is true (caller should publish now).
   641→   641→ */
   642→   642→export async function computeScheduledAt(
   643→   643→  plan: ArticlePlan,
   644→   644→  strategy: SeoStrategyContent | null | undefined,
   645→   645→  publishImmediately: boolean = false
   646→   646→): Promise<Date | null> {
   647→   647→  if (publishImmediately) return null;
   648→   648→
   649→   649→  const now = new Date();
   650→   650→  const oneHourFromNow = new Date(now.getTime() + 60 * 60 * 1000);
   651→   651→
   652→   652→  // 1) پیدا کردن آخرین مقاله زمان‌بندی‌شده (draft با scheduledAt)
   653→   653→  const latestScheduled = await db.article.findFirst({
   654→   654→    where: {
   655→   655→      status: "draft",
   656→   656→      scheduledAt: { not: null },
   657→   657→    },
   658→   658→    orderBy: { scheduledAt: "desc" },
   659→   659→    select: { scheduledAt: true },
   660→   660→  });
   661→   661→
   662→   662→  // 2) محاسبه تاریخ پایه
   663→   663→  // اگر مقاله زمان‌بندی‌شده‌ای وجود دارد، از آن استفاده کن
   664→   664→  // در غیر این صورت، از امروز استفاده کن
   665→   665→  let baseDate = now;
   666→   666→  if (latestScheduled?.scheduledAt) {
   667→   667→    baseDate = new Date(latestScheduled.scheduledAt);
   668→   668→  }
   669→   669→
   670→   670→  // 3) فاصله دقیقاً ۴ روز بین هر مقاله
   671→   671→  // این تضمین می‌کند که مقالات با فاصله منظم ۴ روز منتشر شوند
   672→   672→  // مثلاً ۱۰ مقاله = ۴۰ روز انتشار
   673→   673→  const gapDays = 4;
   674→   674→
   675→   675→  const target = new Date(baseDate);
   676→   676→  target.setDate(target.getDate() + gapDays);
   677→   677→  target.setHours(9, 0, 0, 0); // ساعت ۹ صبح
   678→   678→
   679→   679→  // 4) مطمئن شو که تاریخ در آینده است
   680→   680→  if (target.getTime() < oneHourFromNow.getTime()) {
   681→   681→    // اگر تاریخ محاسبه‌شده در گذشته است، ۲ روز از امروز استفاده کن
   682→   682→    const fallback = new Date(now);
   683→   683→    fallback.setDate(fallback.getDate() + 2);
   684→   684→    fallback.setHours(9, 0, 0, 0);
   685→   685→    return fallback;
   686→   686→  }
   687→   687→
   688→   688→  return target;
   689→   689→}

  1017→  1017→export async function runSeoAgent(opts: {
  1018→  1018→  runId: string;
  1019→  1019→  adminId: string;
  1020→  1020→  mode: "full" | "continue" | "strategy_only";
  1021→  1021→  count: number;
  1022→  1022→  /**
  1023→  1023→   * If true, generated articles are published immediately instead of being
  1024→  1024→   * scheduled for future publishing. Defaults to false (schedule articles).
  1025→  1025→   */
  1026→  1026→  publishImmediately?: boolean;
  1027→  1027→}): Promise<RunContext> {
  1028→  1028→  const ctx: RunContext = {
  1029→  1029→    runId: opts.runId,
  1030→  1030→    mode: opts.mode,
  1031→  1031→    count: opts.count,
  1032→  1032→    logs: [],
  1033→  1033→    adminId: opts.adminId,
  1034→  1034→    startedAt: Date.now(),
  1035→  1035→    successCount: 0,
  1036→  1036→    failCount: 0,
  1037→  1037→    articles: [],
  1038→  1038→    errors: [],
  1039→  1039→    publishImmediately: opts.publishImmediately === true,
  1040→  1040→  };
  1041→  1041→
  1042→  1042→  const startedRun = await db.seoAgentRun.update({
  1043→  1043→    where: { id: opts.runId },
  1044→  1044→    data: {
  1045→  1045→      status: "running",
  1046→  1046→      mode: opts.mode,
  1047→  1047→      requestedCount: opts.count,
  1048→  1048→      startedAt: new Date(),
  1049→  1049→    },
  1050→  1050→  }).catch(() => null);
  1051→  1051→
  1052→  1052→  try {
  1053→  1053→    // Step 1: Always analyze site first (for accurate internal linking)
  1054→  1054→    const site = await analyzeSite(ctx);
  1055→  1055→
  1056→  1056→    // Step 2: Generate/refresh strategy (in full mode) or load existing
  1057→  1057→    let strategy: SeoStrategyContent;
  1058→  1058→    if (opts.mode === "full") {
  1059→  1059→      strategy = await generateStrategy(ctx, site);
  1060→  1060→    } else {
  1061→  1061→      const existing = await db.seoStrategy.findFirst({
  1062→  1062→        where: { isActive: true },
  1063→  1063→        orderBy: { version: "desc" },
  1064→  1064→      });
  1065→  1065→      if (existing) {
  1066→  1066→        strategy = JSON.parse(existing.content) as SeoStrategyContent;
  1067→  1067→        log(ctx, "info", `📋 استفاده از استراتژی موجود (نسخه ${existing.version})`);
  1068→  1068→      } else {
  1069→  1069→        // No strategy yet — must generate
  1070→  1070→        log(ctx, "warn", "استراتژی موجود نیست — تولید استراتژی جدید");
  1071→  1071→        strategy = await generateStrategy(ctx, site);
  1072→  1072→      }
  1073→  1073→    }
  1074→  1074→
  1075→  1075→    if (opts.mode === "strategy_only") {
  1076→  1076→      log(ctx, "success", "🏁 حالت strategy_only — استراتژی تولید شد، بدون تولید مقاله");
  1077→  1077→      await finishRun(ctx, "completed");
  1078→  1078→      return ctx;
  1079→  1079→    }
  1080→  1080→
  1081→  1081→    // Step 3: Plan articles (or use existing planned ones)
  1082→  1082→    let plans: ArticlePlan[] = [];
  1083→  1083→
  1084→  1084→    // First: use existing planned articles (continue mode)
  1085→  1085→    const existingPlans = await db.seoArticlePlan.findMany({
  1086→  1086→      where: { status: "planned" },
  1087→  1087→      orderBy: { priority: "desc" },
  1088→  1088→      take: opts.count,
  1089→  1089→    });
  1090→  1090→
  1091→  1091→    if (existingPlans.length >= opts.count && opts.mode === "continue") {
  1092→  1092→      plans = existingPlans.map((p) => ({
  1093→  1093→        id: p.id,
  1094→  1094→        keyword: p.keyword,
  1095→  1095→        title: p.title,
  1096→  1096→        slug: p.slug,
  1097→  1097→        category: p.category,
  1098→  1098→        excerpt: p.excerpt,
  1099→  1099→        pillar: p.pillar,
  1100→  1100→        secondaryKeywords: p.secondaryKeywords.split(", ").filter(Boolean),
  1101→  1101→        internalLinks: JSON.parse(p.internalLinks || "[]"),
  1102→  1102→        coverImagePrompt: p.coverImagePrompt,
  1103→  1103→        inlineImagePrompts: [],
  1104→  1104→      }));
  1105→  1105→      log(ctx, "info", `📋 استفاده از ${plans.length} مقاله برنامه‌ریزی‌شده موجود در صف`);
  1106→  1106→    } else {
  1107→  1107→      // Plan new articles
  1108→  1108→      const needed = opts.count - existingPlans.length;
  1109→  1109→      if (needed > 0) {
  1110→  1110→        log(ctx, "info", `📅 نیاز به ${needed} مقاله جدید — برنامه‌ریزی...`);
  1111→  1111→        await planArticles(ctx, strategy, site, Math.max(needed, opts.count));
  1112→  1112→      }
  1113→  1113→      // Now fetch all planned
  1114→  1114→      const allPlanned = await db.seoArticlePlan.findMany({
  1115→  1115→        where: { status: "planned" },
  1116→  1116→        orderBy: { priority: "desc" },
  1117→  1117→        take: opts.count,
  1118→  1118→      });
  1119→  1119→      plans = allPlanned.map((p) => ({
  1120→  1120→        id: p.id,
  1121→  1121→        keyword: p.keyword,
  1122→  1122→        title: p.title,
  1123→  1123→        slug: p.slug,
  1124→  1124→        category: p.category,
  1125→  1125→        excerpt: p.excerpt,
  1126→  1126→        pillar: p.pillar,
  1127→  1127→        secondaryKeywords: p.secondaryKeywords.split(", ").filter(Boolean),
  1128→  1128→        internalLinks: JSON.parse(p.internalLinks || "[]"),
  1129→  1129→        coverImagePrompt: p.coverImagePrompt,
  1130→  1130→        inlineImagePrompts: [],
  1131→  1131→      }));
  1132→  1132→    }
  1133→  1133→
  1134→  1134→    if (plans.length === 0) {
  1135→  1135→      log(ctx, "warn", "هیچ مقاله‌ای برای تولید یافت نشد");
  1136→  1136→      await finishRun(ctx, "completed");
  1137→  1137→      return ctx;
  1138→  1138→    }
  1139→  1139→
  1140→  1140→    log(ctx, "info", `🚀 شروع تولید ${plans.length} مقاله`);
  1141→  1141→
  1142→  1142→    // Step 4: Generate each article (sequentially to avoid rate limits)
  1143→  1143→    // Re-analyze site before each article to ensure accurate internal linking (new articles get added)
  1144→  1144→    for (let i = 0; i < plans.length; i++) {
  1145→  1145→      const plan = plans[i];
  1146→  1146→      log(ctx, "info", `━━━ مقاله ${i + 1}/${plans.length} ━━━`);
  1147→  1147→      try {
  1148→  1148→        // Refresh site analysis (so newly-published articles can be linked)
  1149→  1149→        const freshSite = i === 0 ? site : await analyzeSite(ctx);
  1150→  1150→        const result = await generateArticle(ctx, plan, freshSite, strategy);
  1151→  1151→        ctx.articles.push({
  1152→  1152→          id: result.articleId,
  1153→  1153→          title: plan.title,
  1154→  1154→          slug: plan.slug,
  1155→  1155→          keyword: plan.keyword,
  1156→  1156→          coverImage: result.coverUrl ? "✓" : "✗",
  1157→  1157→          words: result.wordCount,
  1158→  1158→          scheduledAt: result.scheduledAt ? result.scheduledAt.toISOString() : null,
  1159→  1159→        });
  1160→  1160→        ctx.successCount++;
  1161→  1161→      } catch (e: any) {
  1162→  1162→        log(ctx, "error", `مقاله "${plan.title}" شکست خورد: ${e.message}`);
  1163→  1163→        ctx.errors.push(`${plan.title}: ${e.message}`);
  1164→  1164→        ctx.failCount++;
  1165→  1165→        if (plan.id) {
  1166→  1166→          await db.seoArticlePlan.update({
  1167→  1167→            where: { id: plan.id },
  1168→  1168→            data: { status: "failed", errorMessage: e.message.slice(0, 500) },
  1169→  1169→          }).catch(() => null);
  1170→  1170→        }
  1171→  1171→      }
  1172→  1172→      // Persist progress to DB after each article
  1173→  1173→      await persistRunProgress(ctx);
  1174→  1174→    }
  1175→  1175→
  1176→  1176→    log(
  1177→  1177→      ctx,
  1178→  1178→      "success",
  1179→  1179→      `🏁 پایان اجرا — ${ctx.successCount} موفق، ${ctx.failCount} ناموفق`
  1180→  1180→    );
  1181→  1181→
  1182→  1182→    await finishRun(ctx, ctx.failCount === 0 ? "completed" : "partial");
  1183→  1183→    return ctx;
  1184→  1184→  } catch (e: any) {
  1185→  1185→    log(ctx, "error", `خطای بحرانی در اجرای ایجنت: ${e.message}`);
  1186→  1186→    ctx.errors.push(`CRITICAL: ${e.message}`);
  1187→  1187→    await finishRun(ctx, "failed");
  1188→  1188→    return ctx;
  1189→  1189→  }
  1190→  1190→}

  1252→  1252→export async function startBackgroundRun(opts: {
  1253→  1253→  adminId: string;
  1254→  1254→  mode: "full" | "continue" | "strategy_only";
  1255→  1255→  count: number;
  1256→  1256→  publishImmediately?: boolean;
  1257→  1257→}): Promise<{ runId: string; alreadyRunning: boolean }> {
  1258→  1258→  if (isAgentRunning()) {
  1259→  1259→    return { runId: getCurrentRunId() || "", alreadyRunning: true };
  1260→  1260→  }
  1261→  1261→  // Create run record
  1262→  1262→  const run = await db.seoAgentRun.create({
  1263→  1263→    data: {
  1264→  1264→      mode: opts.mode,
  1265→  1265→      requestedCount: opts.count,
  1266→  1266→      status: "running",
  1267→  1267→      startedAt: new Date(),
  1268→  1268→    },
  1269→  1269→  });
  1270→  1270→  globalThis.__seoAgentRunning = true;
  1271→  1271→  globalThis.__seoAgentRunId = run.id;
  1272→  1272→
  1273→  1273→  // Fire and forget — run in background
  1274→  1274→  runSeoAgent({
  1275→  1275→    runId: run.id,
  1276→  1276→    adminId: opts.adminId,
  1277→  1277→    mode: opts.mode,
  1278→  1278→    count: opts.count,
  1279→  1279→    publishImmediately: opts.publishImmediately,
  1280→  1280→  })
  1281→  1281→    .catch((e) => {
  1282→  1282→      console.error("[SEO-AGENT] background run crashed:", e);
  1283→  1283→    })
  1284→  1284→    .finally(() => {
  1285→  1285→      globalThis.__seoAgentRunning = false;
  1286→  1286→      globalThis.__seoAgentRunId = undefined;
  1287→  1287→    });
  1288→  1288→
  1289→  1289→  return { runId: run.id, alreadyRunning: false };
  1290→  1290→}
